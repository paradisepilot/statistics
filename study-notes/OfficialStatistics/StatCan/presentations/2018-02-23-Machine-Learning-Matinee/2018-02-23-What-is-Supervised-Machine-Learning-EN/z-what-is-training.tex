\begin{frame}{\Large What is \textit{training}?}

\scriptsize

\pause Want: a machine
\pause to make prediction $\widehat{Y}$
\pause of unknown $Y$
\pause based on observed $X$

\pause
\vskip 0.10cm
Training data:\;
$(x_{1},y_{1})$, $(x_{2},y_{2})$, $(x_{3},y_{3})$, \,$\ldots$\;, $(x_{n},y_{n})$

\small

\pause
\vskip 0.30cm
