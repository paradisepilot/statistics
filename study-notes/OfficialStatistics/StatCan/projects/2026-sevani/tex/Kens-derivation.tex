\subsection{Linear imputation estimators}

\begin{eqnarray*}
\widehat{T}_{Y}
& := &
    \underset{i\,\in\,s}{\sum}\;w_{i}\,\widetilde{Y}_{i}
\\
& = &
    \underset{i\,\in\,s}{\sum}\;w_{i}\left(\,
        \rho_{i}\,Y_{i}
        \overset{{\color{white}.}}{+}
        (1-\rho_{i})\,Y^{*}_{i}
        \,\right)
\;\; = \;\;
    \underset{i\,\in\,s}{\sum}\;w_{i}\,\rho_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,s}{\sum}\;w_{i}\,(1-\rho_{i})\,Y^{*}_{i}
\\
& = &
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,\rho_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,Y^{*}_{i}
\\
& = &
    {\color{red}
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{-}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
    }
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,\rho_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,Y^{*}_{i}
\\
& = &
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{-}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,Y^{*}_{i}
\\
& = &
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
\end{eqnarray*}

\bias\!\left(\;
    \left.

    \Xobs
    \,\right)
& := &
    E\!\left[\;
        \left.
        \widehat{T}_{Y} - T_{Y}

        \,\right]

    E_{YI\rho}\!\left[\;

        \,\right]

    E_{YI\rho}\!\left[\;
        \left.
        \widehat{T}_{Y}
        \,\right\vert\,
        \Xobs
        \,\right]

    E_{YI\rho}\!\left[\;
        \left.
        T_{Y}

        \Xobs
        \,\right]
