\subsection{Linear imputation estimators}

\begin{definition}[Imputation estimator \,$\widetilde{T}_{Y}$]
\mbox{}
\vskip 0.1cm
\noindent
\begin{eqnarray*}
\widetilde{T}_{Y}
& := &
    \underset{i\,\in\,s}{\sum}\;w_{i}\,\widetilde{Y}_{i}
\;\; = \;\;
    \underset{i\,\in\,s}{\sum}\;w_{i}\left(\,
        \rho_{i}\,Y_{i}
        \overset{{\color{white}.}}{+}
        (1-\rho_{i})\,Y^{*}_{i}
        \,\right)
\end{eqnarray*}
\end{definition}

\begin{lemma}[A formula for \,$\widetilde{T}_{Y}$]
\mbox{}
\vskip 0.1cm
\noindent
Assume ...
\begin{eqnarray*}
\widetilde{T}_{Y}
% & := &
%     \underset{i\,\in\,s}{\sum}\;w_{i}\,\widetilde{Y}_{i}
% \;\; = \;\;
%     \underset{i\,\in\,s}{\sum}\;w_{i}\left(\,
%         \rho_{i}\,Y_{i}
%         \overset{{\color{white}.}}{+}
%         (1-\rho_{i})\,Y^{*}_{i}
%         \,\right)
% \\
& = &
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
\end{eqnarray*}
\end{lemma}
\proof
\begin{eqnarray*}
\widetilde{T}_{Y}
& := &
    \underset{i\,\in\,s}{\sum}\;w_{i}\,\widetilde{Y}_{i}
\;\; = \;\;
    \underset{i\,\in\,s}{\sum}\;w_{i}\left(\,
        \rho_{i}\,Y_{i}
        \overset{{\color{white}.}}{+}
        (1-\rho_{i})\,Y^{*}_{i}
        \,\right)
\\
& = &
    \underset{i\,\in\,s}{\sum}\;w_{i}\,\rho_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,s}{\sum}\;w_{i}\,(1-\rho_{i})\,Y^{*}_{i}
\;\; = \;\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,\rho_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,Y^{*}_{i}
\\
& = &
    {\color{red}
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{-}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
    }
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,\rho_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,Y^{*}_{i}
\\
& = &
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{-}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,Y^{*}_{i}
\\
& = &
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
    \;\overset{{\color{white}.}}{+}\;
    \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
\end{eqnarray*}
\qed

\begin{lemma}[Overall conditional bias of \,$\widetilde{T}_{Y}$; see Equation (5.2) in \cite{Beaumont2011} and preceding paragraph]
\label{biasTtilde}
\mbox{}
\vskip 0.1cm
\noindent
Assume that
\,$Y = (Y_{1},\ldots,Y_{N})$\,
and
\,$(I,\rho) = (I_{1},\ldots,I_{N},\rho_{1},\ldots,\rho_{N})$\,
are conditionally independent given
\,$\Xobs$.\,
Then,
\begin{eqnarray*}
\bias\!\left(\;
    \left.
    \widetilde{T}_{Y}
    \;\right\vert\,
    \Xobs
    \,\right)
& = &
    E_{I\rho}\!\left[\;
        \underset{i\,\in\,{\color{red}s_{M}}}{\sum}\,w_{i}\,
        E_{Y}\!\left[\;
            \left.
            Y^{*}_{i} \overset{{\color{white}.}}{-} Y_{i}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right]
        \,\right]
\end{eqnarray*}
\end{lemma}
\begin{remark}
\mbox{}
\vskip -0.1cm
\noindent
\begin{itemize}
\item
    In practice, we therefore would like to choose an imputation strategy such that
    \begin{equation*}
    E_{Y}\!\left[\;
        \left.
        Y^{*}_{i} \overset{{\color{white}.}}{-} Y_{i}
        \;\right\vert\,
        s,s_{R},\Xobs
        \,\right]
    \;\; \approx \;\;
        0,
    \quad
    \textnormal{for each \,$i \in s_{M}$}
    \end{equation*}
    See the paragraph after Equation (5.2), p.175, \cite{Beaumont2011}.
\item
    The conditioning on \,$\Xobs$\, is largely suppressed in \cite{Beaumont2011} for notational brevity.
    We have restored it in our notation in these study notes to help avoid ambiguity.
\end{itemize}
\end{remark}
\proofof Lemma \ref{biasTtilde}
\begin{eqnarray*}
\bias\!\left(\;
    \left.
    \widetilde{T}_{Y}
    \;\right\vert\,
    \Xobs
    \,\right)
& := &
    E\!\left[\;
        \left.
        \widetilde{T}_{Y} - T_{Y}
        \,\right\vert\,
        \Xobs
        \,\right]
\;\; = \;\;
    E_{YI\rho}\!\left[\;
        \left.
        \widetilde{T}_{Y} - T_{Y}
        \,\right\vert\,
        \Xobs
        \,\right]
\;\; = \;\;
    E_{YI\rho}\!\left[\;
        \left.
        \widetilde{T}_{Y}
        \,\right\vert\,
        \Xobs
        \,\right]
    \,-\,
    E_{YI\rho}\!\left[\;
        \left.
        T_{Y}
        \,\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    E_{YI\rho}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
        \;\overset{{\color{white}.}}{+}\;
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
        \,\;\right\vert\,
        \Xobs
        \,\right]
    \,-\,
    E_{YI\rho}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;Y_{i}
        \;\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    E_{YI\rho}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
        \;\right\vert\,
        \Xobs
        \,\right]
    \,+\,
    E_{YI\rho}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
        \;\right\vert\,
        \Xobs
        \,\right]
    \,-\,
    E_{Y}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;Y_{i}
        \;\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    E_{I\rho}\!\left[\,E_{Y}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,Y_{i}
        \;\right\vert\,
        s, s_{R}, \Xobs
        \,\right]\,\right]
    \;-\;
    \underset{i\,\in\,U}{\sum}\;E\!\left[\;Y_{i}\,\vert\,\Xobs\,\right]
\\
&&
    +\;
    E_{YI\rho}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
        \;\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    E_{I\rho}\!\left[\,
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,
        E_{Y}\!\left[\,\left.Y_{i}\,\right\vert\Xobs\,\right]
        \,\right]
    \;-\;
    \underset{i\,\in\,U}{\sum}\;E\!\left[\;Y_{i}\,\vert\,\Xobs\,\right]
    % \,+\,
    % E_{YI\rho}\!\left[\;
    %     \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
    %     \,\right]
\\
&&
    +\;
    E_{YI\rho}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
        \;\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    \underset{i\,\in\,U}{\sum}\;E_{Y}\!\left[\,\left.Y_{i}\,\right\vert\Xobs\,\right]
    \;-\;
    \underset{i\,\in\,U}{\sum}\;E\!\left[\;Y_{i}\,\vert\,\Xobs\,\right]
    \;+\;
    E_{YI\rho}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
        \;\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    E_{YI\rho}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
        \;\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    E_{I\rho}\!\left[\;
    E_{Y}\!\left[\;
        \left.
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,(Y^{*}_{i}-Y_{i})
        \;\right\vert\,
        s,s_{R},\Xobs
        \,\right]
        \,\right]
\\
& = &
    E_{I\rho}\!\left[\;
        \underset{i\,\in\,U}{\sum}\;I_{i}\,w_{i}\,(1-\rho_{i})\,
        E_{Y}\!\left[\;
            \left.
            Y^{*}_{i} \overset{{\color{white}.}}{-} Y_{i}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right]
        \,\right]
\\
& = &
    E_{I\rho}\!\left[\;
        \underset{i\,\in\,{\color{red}s_{M}}}{\sum}\,w_{i}\,
        E_{Y}\!\left[\;
            \left.
            Y^{*}_{i} \overset{{\color{white}.}}{-} Y_{i}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right]
        \,\right]
\end{eqnarray*}
\qed

\begin{eqnarray*}
\MSE\!\left(\;
    \left.
    \widetilde{T}_{Y}
    \;\right\vert\,
    \Xobs
    \,\right)
& := &
    E\!\left[\,
        \left.
        (\,\widetilde{T}_{Y} - T_{Y}\,)^{2}
        \,\right\vert\,
        \Xobs
        \,\right]
\;\; = \;\;
    E_{YI\rho}\!\left[\,
        \left.
        (\,\widetilde{T}_{Y} - T_{Y}\,)^{2}
        \,\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    E_{YI\rho}\!\left[\,
        \left.
        (\,\widetilde{T}_{Y} \;{\color{red}-\; \widehat{T}_{Y} \;+\; \widehat{T}_{Y}} \,-\, T_{Y}\,)^{2}
        \,\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    E_{YI\rho}\!\left[\,
        \left.
        (\,\widetilde{T}_{Y} - \widehat{T}_{Y}\,)^{2}
        \,+\,
        2\,(\,\widetilde{T}_{Y} \,-\, \widehat{T}_{Y}\,)(\,\widehat{T}_{Y} \,-\, T_{Y}\,)
        \,+\,
        (\,\widehat{T}_{Y} - T_{Y}\,)^{2}
        \,\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    E_{YI\rho}\!\left[\,
        \left.
        (\,\widehat{T}_{Y} - T_{Y}\,)^{2}
        \,\right\vert\,
        \Xobs
        \,\right]
    \;+\;
    E_{YI\rho}\!\left[\,
        \left.
        (\,\widetilde{T}_{Y} - \widehat{T}_{Y}\,)^{2}
        \,\right\vert\,
        \Xobs
        \,\right]
\\
&&
    +\;
    2\,E_{YI\rho}\!\left[\,
        \left.
        (\,\widetilde{T}_{Y} \,-\, \widehat{T}_{Y}\,)(\,\widehat{T}_{Y} \,-\, T_{Y}\,)
        \,\right\vert\,
        \Xobs
        \,\right]
\\
& = &
    E_{Y}\!\left[\,
    E_{I\rho}\!\left[\,
        \left.
        (\,\widehat{T}_{Y} - T_{Y}\,)^{2}
        \,\right\vert\,
        \Xobs
        \,\right]
        \,\right]
    \;+\;
    E_{I\rho}\!\left[\,
    E_{Y}\!\left[\,
        \left.
        (\,\widetilde{T}_{Y} - \widehat{T}_{Y}\,)^{2}
        \,\right\vert\,
        s,s_{R},\Xobs
        \,\right]
        \,\right]
\\
&&
    +\;2\,
    E_{I\rho}\!\left[\,
    E_{Y}\!\left[\,
        \left.
        (\,\widetilde{T}_{Y} \,-\, \widehat{T}_{Y}\,)(\,\widehat{T}_{Y} \,-\, T_{Y}\,)
        \,\right\vert\,
        s,s_{R},\Xobs
        \,\right]
        \,\right]
\end{eqnarray*}

          %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

The first term above
\,--\,
i.e.,
$
E_{Y}\!\left[\,
E_{I\rho}\!\left[\,
    \left.
    (\,\widehat{T}_{Y} - T_{Y}\,)^{2}
    \,\right\vert\,
    \Xobs
    \,\right]
    \,\right]
$
\,--\,
is called the \textit{sampling variance} in \cite{Beaumont2011}.
First, note that
\begin{eqnarray*}
E_{Y}\!\left[\,
E_{I\rho}\!\left[\,
    \left.
    (\,\widehat{T}_{Y} - T_{Y}\,)^{2}
    \,\right\vert\,
    \Xobs
    \,\right]
    \,\right]
& = &
    E_{Y}\!\left[\,
    \Var_{I\rho}\!\left(\,
        \left.
        \widehat{T}_{Y}
        \,\right\vert\,
        \Xobs
        \,\right)
        \,\right]
\end{eqnarray*}
It is asserted that
\begin{eqnarray*}
E_{Y}\!\left[\,
E_{I\rho}\!\left[\,
    \left.
    (\,\widehat{T}_{Y} - T_{Y}\,)^{2}
    \,\right\vert\,
    \Xobs
    \,\right]
    \,\right]
& \approx &
    E_{Y}\!\left[\,
    E_{I\rho}\!\left[\,
        \left.
        (\,\widehat{T}_{Y} - T_{Y}\,)^{2}
        \,\right\vert\,
        s,s_{R},\mathbf{Y}_{R},\Xobs
        \,\right]
        \,\right]
\end{eqnarray*}
See \cite{Beaumont2009} as well as Equation (5.5), p.175, \cite{Beaumont2011}.
