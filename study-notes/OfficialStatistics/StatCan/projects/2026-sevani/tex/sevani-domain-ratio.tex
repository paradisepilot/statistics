\section{Variance estimation methodology for composite linear imputation estimators for domain ratios}

\subsection{Probabilistic framework of SEVANI \cite{Beaumont2011}}
\label{DomainRatioProbabilisticFramework}
\setcounter{theorem}{0}

        %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

\noindent
\textbf{Probability spaces}
\begin{itemize}
\item
    Let
    \,$U \,=\, \{1,2,\ldots,N\}$.\,
    Let
    \,$\mathcal{P}(U) \,=\, \mathcal{P}(\{1,2,\ldots,N\})$\,
    denote the power set (set of all subsets) of \,$U$.\,
    Let
    \,$\left(\,\mathcal{P}(U)\,,\mathcal{P}(\mathcal{P}(U))\,,\overset{{\color{white}-}}{\pi}\,\right)$\,
    be the probability space induced by a sampling design, i.e.,
    a probability mass function defined on
    \,$\mathcal{P}(U)$.\,
\item
    Let \,$\left(\,\Omega,\mathscr{A},\lambda\,\right)$\, be a probability space.
\end{itemize}

\vskip 0.2cm
\noindent
\textbf{Random variables}
\begin{itemize}
\item
    $\Ydot \,=\, (Y_{1},Y_{2},\ldots,Y_{N})^{T} : \Omega \longrightarrow \Re^{N \times 1}$\,
    is an $\Re^{N \times 1}$-valued random variable defined on \,$\Omega$,\,
    i.e., $\mathscr{A}$-measurable map.
\item
    $\Zdot \,=\, (Z_{1},Z_{2},\ldots,Z_{N})^{T} : \Omega \longrightarrow \Re^{N \times 1}$\,
    is an $\Re^{N \times 1}$-valued random variable defined on \,$\Omega$.\,
\item
    $\Xdotdot\,=\,\left(\,\overset{{\color{white}.}}{X_{ij}}\,\right) : \Omega \longrightarrow \Re^{N \times r}$\,
    is an $\Re^{N \times r}$-valued random variable defined on \,$\Omega$.
\item
    $\Sdot \,=\, (S_{1},S_{2},\ldots,S_{N})^{T} : \mathcal{P}(U) \longrightarrow \{0,1\}^{N \times 1}$\,
    is a $\{0,1\}^{N \times 1}$-valued random variable defined on \,$\mathcal{P}(U)$,\,
    i.e., $\mathcal{P}(\mathcal{P}(U))$-measurable map.
\item
    $\Rdot \,=\, (R_{1},R_{2},\ldots,R_{N})^{T} : {\color{red}\Omega \otimes \mathcal{P}(U)} \longrightarrow \{0,1\}^{N \times 1}$\,
    is a $\{0,1\}^{N \times 1}$-valued random variable defined on \,$\Omega \times \mathcal{P}(U)$,\,
    i.e., $\left(\mathscr{A} \overset{{\color{white}.}}{\times} \mathcal{P}(\mathcal{P}(U))\right)$-measurable map
    that satisfies:
    % \begin{equation*}
    % \mu\!\left(\left\{\;
    %     \omega \in \Omega
    %     \;\left\vert\;
    %     S_{k}(s) = 0
    %     \,\overset{{\color{white}1}}{\Longrightarrow}\,
    %     R_{k}(\omega,s_{R}) = 0
    %     \right.
    %     \;\right\}\right)
    % \; = \;
    %     1\,,
    % \quad
    % \textnormal{for each \,$k \in U$,\, and each \,$s_{R} \subset s \subset U$}
    % \end{equation*}
    % \begin{equation*}
    % (\,\mu\times\pi\,)\!\left(\left\{\;
    %     (\omega,s) \in \Omega
    %     \;\left\vert\;
    %     \begin{array}{c}
    %     S_{k}(s) = 0, \,\textnormal{and}\; R_{k}(\omega,s) = 1,
    %     \\
    %     \overset{{\color{white}.}}{\textnormal{for one or more $k \in U$}}
    %     \end{array}
    %     \right.
    %     \;\right\}\right)
    % \; = \;
    %     0
    % \end{equation*}
    \begin{equation*}
    (\,\mu\times\pi\,)\!\left(\;\,
        \underset{k \in U}{\bigcup}\left\{\;
            (\omega,s) \in \Omega\times\mathcal{P}(U)
            \;\left\vert
            \begin{array}{c}
            S_{k}(s) = 0, %\,\textnormal{and}\;
            \\
            \overset{{\color{white}.}}{R_{k}(\omega,s) = 1}
            \end{array}
            \right.
            \!\!\right\}
        \;\right)
    \; = \;
        0
    \end{equation*}
    Note that this condition stipulates that,
    whenver a population unit \,$k \in U$\, is not selected in the (random) sample \,$s \subset U$,\,
    then it is not a respondent either.
\end{itemize}

\vskip 0.2cm
\noindent
\textbf{Interpretation}
\begin{itemize}
\item
    $U = \{1,2,\ldots,N\}$
    \,--\,
    index set of the target (finite) population of interest,
    and \,$N$\, is the (finite) size of the target population.
\item
    $\left(\,\Omega,\mathscr{A},\lambda\,\right)$
    \,--\,
    underlying source of stochasticity of the imputation model (or superpopulation model).
\item
    $\Ydot$
    \,--\,
    numerator of the $\Re$-valued target variable (population characteristic).
\item
    $\Zdot$
    \,--\,
    denominator of the $\Re$-valued target variable (population characteristic).
\item
    $\Xdotdot$
    \,--\,
    $\Re^{r}$-valued auxiliary variable.
\item
    $\Sdot$
    \,--\,
    $\{0,1\}$-valued binary variable indicating, for each \,$s \subset U$,\,
    whether a population unit \,$k \in U$\, is selected in \,$s$.\,
    More precisely, for each \,$s \subset U$,\, we have:
    \begin{equation*}
    S_{k}(s)
    \;\; = \;\;
        \left\{\begin{array}{cl}
            1\,, & \textnormal{if \,$k \in s$},
            \\
            0\,, & \textnormal{otherwise}.
            \end{array}\right.
    \end{equation*}
\item
    $\Rdot$
    \,--\,
    $\{0,1\}$-valued binary variable indicating, for almost every outcome
    \,$\omega \in \Omega$\,
    and each
    \,$(s,s_{R}) \in \mathcal{P}(U) \times \mathcal{P}(U)$\,
    satisfying
    \,$s_{R} \subset s$,\,
    whether a selected population unit \,$ k \in s$\,
    is a respondent or not.
    % More precisely, for each \,$s \subset U$ and each \,$k \in s$,\, we have:
    % \begin{equation*}
    % R_{k}(\omega,s_{R})
    % \;\; = \;\;
    %     \left\{\begin{array}{cl}
    %         1\,, & \textnormal{if \,$k \in s_{R}$},
    %         \\
    %         0\,, & \textnormal{otherwise}.
    %         \end{array}\right.
    % \end{equation*}
\end{itemize}

\vskip 0.2cm
\noindent
\textbf{Probabilistic assumptions}
\begin{itemize}
\item
    $\pi_{i} \,:=\, P(\,S_{i} = 1\,) \,:=\, \pi\!\left(\,\left\{\,\left. s \overset{{\color{white}.}}{\subset} U \,\right\vert\, i \in s\,\right\}\,\right) \,>\, 0$,\,
    for each \,$i \in U$,\,
    and
    \,$w_{i} \,:=\, \dfrac{1}{P(\,S_{i} = 1\,)}$,\, for each \,$i \in U$.
\item
    $\left(\,\Ydot\,,\Zdot\,,\Xdotdot\,\right)$\, and \,$\Sdot$\, are (unconditionally) independent.
\item
    % $\left(\,\Ydot\,,\Zdot\,\right)$\,
    $\Ydot$\,
    and
    \,$(\,\Sdot\,,\Rdot\,)$\,
    are {\color{red}conditionally independent} given \,$\left(\,\Zdot\,,\Xdotdot\,\right)$.
\item
    $\Cov\!\left(\;Y_{k}\,,\,Y_{l}\;\left\vert\;\Zdot\,,\Xdotdot\right.\right) \,=\, 0$,\,
    for each \,$k, l \in U$\, with \,$k \neq l$.
\end{itemize}

\vskip 0.2cm
\noindent
\textbf{(Non-random) parameters}
\begin{itemize}
\item
    $\dbullet \,=\, (d_{1},d_{2},\ldots,d_{N}) \in \{0,1\}^{N}$\,
    is a (non-random, binary) domain membership indicator.
\end{itemize}

\subsection{Imputation ratio estimator}
\setcounter{theorem}{0}

        %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

\vskip 0.2cm
\begin{definition}[Imputation ratio estimator \,$\widetilde{R}_{Y/Z}$]
\mbox{}
\vskip 0.1cm
\noindent
An \textbf{imputation ratio estimator} for the domain ratio
\begin{equation*}
R_{Y/Z}
\;\; := \;\;
    \dfrac{ T_{Y} }{ \overset{{\color{white}.}}{T_{Z}} }
\;\; = \;\;
    \dfrac{
        \underset{i \in U}{\sum}\,d_{i}Y_{i}
        }{
        \overset{{\color{white}.}}{\underset{i \in U}{\sum}\,d_{i}Z_{i}}
        }
\end{equation*}
is the random variable of the form:
\begin{eqnarray*}
{\color{red}\widetilde{R}_{Y/Z}}
& = &
    \widetilde{R}_{Y/Z}(\,
        \Sdot\,,\,
        \Rdot\,,\,
        \Ydot\,,\,
        \Zdot\,,\,
        \Xdotdot\,;\,
        \dbullet
        \,)
\;\; := \;\;
    {\color{red}
    \dfrac{
        {\color{white}.}
        \widetilde{T}_{Y}
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\widehat{T}_{Z}}
        }
    }
\\
& = &
    \dfrac{
        \underset{i\,\in\,s}{\sum}\;d_{i}\,w_{i}\,\widetilde{Y}_{i}
        }{
        \overset{{\color{white}.}}{\underset{i\,\in\,s}{\sum}\;d_{i}\,w_{i}\,Z_{i}}
        }
\;\; = \;\;
    \dfrac{
        \underset{i\,\in\,U}{\sum}\;d_{i}\,w_{i}\,S_{i}\left(\,
            R_{i}\,Y_{i}
            \overset{{\color{white}.}}{+}
            (1-R_{i})\,Y^{*}_{i}
            \,\right)
        }{
        \overset{{\color{white}.}}{\underset{i\,\in\,U}{\sum}\;d_{i}\,w_{i}\,S_{i}\,Z_{i}}
        }
\end{eqnarray*}
where
\;$\widetilde{Y}_{i} \,=\, R_{i}\,Y_{i}+(1-R_{i})\,Y^{*}_{i}$\;
and
\;$Y^{*}_{i} \,=\, Y^{*}_{i}(\,\Sdot,\Rdot,\Ydot,\Xdotdot\,)$.\,
\end{definition}

\begin{lemma}[Decomposition of \,$\MSE(\,\widetilde{R}_{Y/Z}\,\vert\,\Zdot\,,\Xobs\,)$; see Equation (5.3) in \cite{Beaumont2011}]
\label{DecompositionMSETtilde}
\mbox{}
\vskip 0.1cm
\noindent
Let
\begin{eqnarray*}
\widetilde{R}_{Y/Z}
& = &

\begin{lemma}[Approximately unbiased estimator for sampling variance component for ratio estimators]
\label{RatioEstimatorSamplingVariance}
\mbox{}
\vskip -0.1cm
\noindent
Let
\begin{eqnarray*}
\widetilde{R}_{Y/Z}
& = &
    \widetilde{R}_{Y/Z}(\,
        \Sdot\,,\,
        \Rdot\,,\,
        \Ydot\,,\,
        \Zdot\,;\,
        \Xdotdot\,,\,
        \dbullet
        \,)
\;\;\; := \;\;\;
    \dfrac{
        \underset{i\,\in\,s}{\sum}\;d_{i}\,w_{i}\,\widetilde{Y}_{i}
        }{
        \overset{{\color{white}.}}{
            \underset{i\,\in\,s}{\sum}\;d_{i}\,w_{i}\,Z_{i}
            }
        }
\;\; = \;\;
    \dfrac{
        \underset{i\,\in\,U}{\sum}\;d_{i}\,w_{i}\,S_{i}\left(\,
            R_{i}\,Y_{i}
            \overset{{\color{white}.}}{+}
            (1-R_{i})\,Y^{*}_{i}
            \,\right)
        }{
        \overset{{\color{white}.}}{
            \underset{i\,\in\,U}{\sum}\;d_{i}\,w_{i}\,S_{i}\,Z_{i}
            }
        }
\end{eqnarray*}
be an imputation estimator for the domain ratio
\;$
R_{Y/Z}
\, := \,
    \dfrac{T_{Y}}{\overset{{\color{white}.}}{T_{Z}}}
\, = \,
    \dfrac{
        {\color{white}.}
        \underset{i \in U}{\sum}\;d_{i}\,Y_{i}
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\underset{i \in U}{\sum}\;d_{i}\,Z_{i}}
        }
$,\;
where
\begin{equation*}
\widetilde{Y}_{i} \;=\; R_{i}\,Y_{i}+(1-R_{i})\,Y^{*}_{i}
\quad\;\;\textnormal{and}\quad\;
Y^{*}_{i} \;=\; Y^{*}_{i}(\,\Sdot,\Rdot,\Ydot,\Xdotdot\,)
\end{equation*}
\vskip 0.2cm
\noindent
Suppose the quantities
\,$
\Sdot\,,\,
\Rdot\,,\,
\Ydot\,,\,
\Zdot\,,\,
\Xdotdot\,,\,
\dbullet
$\,
satisfy the probabilistic assumptions specified in Subsection \ref{DomainRatioProbabilisticFramework}.
\vskip 0.2cm
\noindent
Then, the following statements are true:
\renewcommand{\theenumi}{\roman{enumi}}
\renewcommand{\labelenumi}{\mbox{}\;\;\,(\theenumi)$\quad$}
\begin{enumerate}
\item
    If
    \;$v(\,s\,;y_{\bullet},\zdot\,)$\,
    is an (arbitrary) design-unbiased estimator for the sampling variance, i.e.,
    \begin{equation*}
    E_{S}\!\left[\;
        v(\,s\,;\Ydot,\Zdot\,)
        \,\left\vert\;
        \Ydot,\Zdot\,,\Xobs
        \right.
        \,\right]
    \;\; = \;\;
        \Var_{S}\!\left(\,
            \left.
            \widehat{R}_{Y/Z}
            \,\right\vert\,
            \Ydot,
            \Zdot,
            \Xobs
            \,\right),
    \end{equation*}
    then
    \,$
    E_{Y_{M}}\!\left[\,
        v(\,s\,;\Ydot,\Zdot\,)
        \left\vert\,
        s,s_{R},\YsubR,\Zdot\,,\Xobs
        \right.
        \right]
    $\,
    is an $(\Sdot,\Rdot,\YsubR)$-unbiased estimator for
    \,$
    E_{Y}\!\left[\,
        \Var_{S}\!\left(\,
            \left.
            \widehat{T}_{Y}
            \,\right\vert\,
            \Ydot,
            \Zdot,
            \Xobs
            \,\right)
    \,\right]
    $,\,
    i.e.,
    \begin{equation*}
    E_{SR}\,
    E_{Y_{R}}\,
    E_{Y_{M}}\!\left[\,
        v(\,s\,;\Ydot,\Zdot\,)
        \,\left\vert\;
        s,s_{R},\YsubR,\Zdot\,,\Xobs
        \right.
        \,\right]
    \;\; = \;\;
        E_{Y}\!\left[\,
            \Var_{S}\!\left(\,
                \left.
                \widehat{T}_{Y}
                \,\right\vert\,
                \Ydot,
                \Zdot,
                \Xobs
                \,\right)
            \,\right]
    \end{equation*}
\item
    If we choose in turn to approximate
    \,$v(\,s\,;y_{\bullet}\,)$\,
    by the {\color{red}approximately} design-unbiased linearization-based estimator
    \,$v_{L(Y/Z)}(\,s\,;y_{\bullet},\zdot\,)$\,
    for the sampling variance (see Result 5.6.2, p.178, \cite{Sarndal1992}), i.e.,
    \begin{eqnarray*}
    &&
        v_{L(Y/Z)}(\,s\,;y_{\bullet},\zdot\,)
    \\
    & := &
        \dfrac{1}{
            \overset{{\color{white}.}}{\widehat{T}(s\,;\zdot)^{2}}
            }
        \cdot
        \underset{k \in s}{\sum}\,
        \underset{l \in s}{\sum}\;
        \dfrac{\pi_{kl} - \pi_{k}\pi_{l}}{\pi_{kl}}
        \left(
            \dfrac{
                d_{k}\left(\,y_{k}-\widehat{R}_{Y/Z}(s\,;\ydot,\zdot) \cdot z_{k}\,\right)
                }{
                \pi_{k}
                }
            \right)
        \left(
            \dfrac{
                d_{l}\left(\,y_{l}-\widehat{R}_{Y/Z}(s\,;\ydot,\zdot) \cdot z_{l}\,\right)
                }{
                \pi_{l}
                }
            \right)
    \\
    & = &
        \dfrac{1}{
            \overset{{\color{white}.}}{\widehat{T}(s\,;\zdot)^{2}}
            }
        \cdot
        \underset{k \in s}{\sum}\,
        \underset{l \in s}{\sum}\;
        \dfrac{\pi_{kl} - \pi_{k}\pi_{l}}{\pi_{kl}}\,
        \left(\,
            \dfrac{d_{k}}{\pi_{k}}
            \left(\,
                y_{k}
                \, - \,
                \dfrac{
                    \widehat{T}(s\,;\ydot)
                    }{
                    \overset{{\color{white}.}}{\widehat{T}(s\,;\zdot)}
                    }
                \cdot
                z_{k}
                \,\right)
            \right)
        \left(\,
            \dfrac{d_{l}}{\pi_{l}}
            \left(\,
                y_{l}
                \, - \,
                \dfrac{
                    \widehat{T}(s\,;\ydot)
                    }{
                    \overset{{\color{white}.}}{\widehat{T}(s\,;\zdot)}
                    }
                \cdot
                z_{l}
                \,\right)
            \right)
    \\
    & = &
        \dfrac{
            v_{\textnormal{HT}}(\,s\,;\ydot\,)
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)^{2}
                }
            }
        \; + \;
        \dfrac{
            (-2)\left.\widehat{T}(s\,;\ydot)\right\slash\widehat{T}(s\,;\zdot)
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)^{2}
                }
            }
        \left(\,
            \underset{k \in s}{\sum}\;
            \underset{l \in s}{\sum}
                \left(\dfrac{\pi_{kl}-\pi_{k}\pi_{l}}{\pi_{kl}}\right)
                \dfrac{d_{k}\,y_{k}}{\pi_{k}}\,
                \dfrac{d_{l}\,z_{l}}{\pi_{l}}
            \right)
    \\
    &&
        \; + \;
        \dfrac{
            \left.
            \widehat{T}(s\,;\ydot)^{2}
            \right\slash
            \widehat{T}(s\,;\zdot)^{2}
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)^{2}
                }
            }
        \cdot
        v_{\textnormal{HT}}(s\,;\zdot)
    \end{eqnarray*}
    then
    \,$
    E_{Y_{M}}\!\left[\,
        v_{L(Y/Z)}(\,s\,;\Ydot,\Zdot\,)
        \left\vert\,
        s,s_{R},\YsubR = y_{R},\Zdot\,,\Xobs
        \right.
        \right]
    $\,
    can be expressed as follows:
    \begin{eqnarray*}
    &&
        E_{Y_{M}}\!\left[\;
            v_{L(Y/Z)}(\,s\,;\Ydot,\Zdot\,)
            \,\left\vert\,
            \begin{array}{c}
                s,s_{R},\\
                \YsubR = y_{R},\\
                \Zdot\,,\Xobs
                \end{array}
            \right.
            \!\right]
    \\
    & = &
        {\color{blue}
        v_{L(Y/Z)}(\,s\,;\widetilde{y}_{\bullet},\zdot\,)
        \; + \;
        \dfrac{
            1
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)^{2}
                }
            }
        \left(\,
            \underset{k \in s_{M}}{\sum}\;
            d_{k}
            \left(1-\dfrac{1}{w_{k}}\right)
            w_{k}^{2}\,\sigma_{k}^{2}
            \,\right)
        }
    \\
    &&
        \; + \;
        \dfrac{
            (-2)
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)^{3}
                }
            }
        \left(\,
            \underset{k \in {\color{red}s_{M}}}{\sum}\;
            \underset{l \in s}{\sum}
                \left(\dfrac{\pi_{kl}-\pi_{k}\pi_{l}}{\pi_{kl}}\right)
                \dfrac{d_{k}^{\,2}\,\sigma^{2}_{k}}{\pi_{k}^{2}}\,
                \dfrac{d_{l}\,z_{l}}{\pi_{l}}
            \right)
        \; + \;
        \dfrac{
            \underset{k \in s_{M}}{\sum}\,d_{k}^{\,2}\sigma^{2}_{k}/\pi_{k}^{2}
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)^{2}
                }
            }
        \cdot
        \dfrac{
            v_{\textnormal{HT}}(s\,;\zdot)
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)^{2}
                }
            }
    \end{eqnarray*}
    where
    \,$\widetilde{y}_{\bullet}$\,
    is given by:
    \begin{equation*}
    \widetilde{y}_{k}
    \;\; := \;\;
        \left\{\begin{array}{cl}
            y_{k}, & \textnormal{for \,$k \,\in\, s_{R}$}
            \\
            % \overset{{\color{white}1}}{E_{Y}\!\left[\;Y_{k}\,\left\vert\;\Xobs\right.\right]}, & \textnormal{for \,$k \,\in\, s_{M} \,:=\, s \,\backslash\, s_{R}$}
            \overset{{\color{white}\textnormal{\large1}}}{\mu_{k}}, & \textnormal{for \,$k \,\in\, s_{M} \,:=\, s \,\backslash\, s_{R}$}
            \end{array}\right.,
    \end{equation*}
    and, for \,$k \in s_{M}$,\, the parameters
    \,$\mu_{k}$\, and \,$\sigma_{k}^{2}$\,
    are given by:
    \begin{eqnarray*}
    \mu_{k}
    & := &
        {\color{white}...}E_{Y}\!\left[\;Y_{k}\,\left\vert\;\Zdot\,,\Xobs\right.\right]
    \\
    \sigma_{k}^{2}
    & := &
        \Var_{Y}\!\left[\;
            Y_{k}
            \,\left\vert\;
            \Zdot\,,\Xobs
            \right.
            \right]
    \end{eqnarray*}
\end{enumerate}
\end{lemma}
\proof
\renewcommand{\theenumi}{\roman{enumi}}
\renewcommand{\labelenumi}{\mbox{}\;\;\,(\theenumi)$\quad$}
\begin{enumerate}
\item
    We simply compute:
    \begin{eqnarray*}
    &&
        E_{SR}\,
        E_{Y_{R}}\,
        E_{Y_{M}}\!\left[\,
            v(\,s\,;\Ydot,\Zdot\,)
            \,\left\vert\;
            s,s_{R},\YsubR,\Zdot\,,\Xobs
            \right.
            \,\right]
    \\
    & = &
        E_{SR}\,
        E_{Y}\!\left[\,
            v(\,s\,;\Ydot,\Zdot\,)
            \,\left\vert\;
            s,s_{R},\Zdot\,,\Xobs
            \right.
            \,\right]
    \\
    & = &
        E_{Y}\,
        E_{SR}\!\left[\,
            v(\,s\,;\Ydot,\Zdot\,)
            \,\left\vert\;
            \Ydot,\Zdot\,,\Xobs
            \right.
            \,\right]
    \\
    & = &
        E_{Y}\,
        E_{S}\!\left[\,
            v(\,s\,;\Ydot,\Zdot\,)

\begin{lemma}[Estimator for nonresponse variance -- case of domain ratio]
\label{DomainRatioNonresponseVariance}
\mbox{}
\vskip 0.1cm
\noindent
Let
\begin{eqnarray*}
\widetilde{R}_{Y/Z}
& = &
    \widetilde{R}_{Y/Z}(\,
        \Sdot\,,\,
        \Rdot\,,\,
        \Ydot\,,\,
        \Zdot\,;\,
        \Xdotdot\,,\,
        \dbullet
        \,)
\;\;\; := \;\;\;
    \dfrac{
        \underset{i\,\in\,s}{\sum}\;d_{i}\,w_{i}\,\widetilde{Y}_{i}
        }{
        \overset{{\color{white}.}}{
            \underset{i\,\in\,s}{\sum}\;d_{i}\,w_{i}\,Z_{i}
            }
        }
\;\; = \;\;
    \dfrac{
        \underset{i\,\in\,U}{\sum}\;d_{i}\,w_{i}\,S_{i}\left(\,
            R_{i}\,Y_{i}
            \overset{{\color{white}.}}{+}
            (1-R_{i})\,Y^{*}_{i}
            \,\right)
        }{
        \overset{{\color{white}.}}{
            \underset{i\,\in\,U}{\sum}\;d_{i}\,w_{i}\,S_{i}\,Z_{i}
            }
        }
\end{eqnarray*}
be an imputation estimator for the domain ratio
\;$
R_{Y/Z}
\, := \,
    \dfrac{T_{Y}}{\overset{{\color{white}.}}{T_{Z}}}
\, = \,
    \dfrac{
        {\color{white}.}
        \underset{i \in U}{\sum}\;d_{i}\,Y_{i}
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\underset{i \in U}{\sum}\;d_{i}\,Z_{i}}
        }
$,\;
where
\,$\widetilde{Y}_{i} \;=\; R_{i}\,Y_{i}+(1-R_{i})\,Y^{*}_{i}$,\,
and
\begin{equation*}
Y^{*}_{k}
\;\; = \;\;
    Y^{*}_{k}(\,\Sdot,\Rdot,\Ydot,\Xdotdot\,)
\;\; = \;\;
    \varphi^{(j)}_{0k}
    \,+\,
    \underset{l \in s_{R}}{\sum}\,\varphi^{(j)}_{lk}\,Y_{l}\,,
    \quad
    \textnormal{for each \,$k \in s_{M} := s \,\backslash\, s_{R}$}
\end{equation*}
\vskip 0.1cm
\noindent
Define:
\,$
\mu_{k} \;:=\, E_{Y}\!\left[\;Y_{k}
    \,\left\vert\;
    \Zdot\,,\Xobs
    \right.
    \,\right]
$,\,
\,$
\sigma_{k}^{2} \;:=\, \Var_{Y}\!\left[\;Y_{k}
    \,\left\vert\;
    \Zdot\,,\Xobs
    \right.
    \,\right]
$,\,
\begin{equation*}
W^{(+)}_{{\color{red}0}}
\;\; := \;\;
    \left(\;
        \overset{J}{\underset{j=1}{\sum}}\;
        \underset{l \in s_{M}^{(j)}}{\sum}\,d_{l}\,w_{l}\,\varphi^{(j)}_{{\color{red}0}l}
        \,\right),
\quad\;\;\textnormal{and}\quad\;\;
W^{(+)}_{{\color{red}l}}
\;\; := \;\;
    \left(\;
        \overset{J}{\underset{j=1}{\sum}}\;
        \underset{k \in s_{M}^{(j)}}{\sum}\,d_{k}\,w_{k}\,\varphi^{(j)}_{{\color{red}l}k}
        \,\right),
\;\;
\textnormal{for each \,$l \in s_{R}$}
\end{equation*}
\vskip 0.2cm
\noindent
% Suppose the quantities
% \,$
% \Sdot\,,\,
% \Rdot\,,\,
% \Ydot\,,\,
% \Xdotdot\,,\,
% \dbullet
% $\,
% satisfy the probabilistic assumptions specified in Subsection \ref{ProbabilisticFramework}.

\Rdot\,,\,
\Ydot\,,\,

\dbullet
$\,

Then, the following statements are true:
\renewcommand{\theenumi}{\roman{enumi}}
\renewcommand{\labelenumi}{\mbox{}\;\;\,(\theenumi)$\quad$}
\begin{enumerate}
\item
    $
    E_{Y}\!\left[\;
        (\,\widetilde{R}_{Y/Z} - \widehat{R}_{Y/Z}\,)^{2}

            \\

            \end{array}

        \!\right]

            \\
            \Zdot\,,\Xobs

        \!\right]
        \right]

    E_{Y}\!\left[\,

        \begin{array}{c}
            s,s_{R},
            \\
            \Zdot\,,\Xobs
            \end{array}
        \right.
        \!\right]

    &&

    \end{eqnarray*}
\end{enumerate}
\end{lemma}
\proof

\item

        \cdots
    \; = \;

    \end{equation*}
    This completes the proof of Claim 1.

    \vskip 0.5cm

        \!\right]

        \left(\;

                \Zdot\,,\Xobs
                \end{array}
            \!\right]

            \left.

                \widetilde{T}_{Y}

                \widehat{T}_{Y}
            \,\left\vert
            \begin{array}{c}
                s,s_{R},
                \\
                \Zdot\,,\Xobs
                \end{array}
            \right.
            \!\right]
    \\
    & = &
        \left(\,\dfrac{1}{ \overset{{\color{white}.}}{\widehat{T}(s\,;\zdot)} }\,\right)^{\!2}
        \cdot
        \Var_{Y}\!\left[\,
            \left.
            W^{(+)}_{0}
            \,+\,
            \underset{l \in s_{R}}{\sum}\,W^{(+)}_{l}\,Y_{l}
            \,-\,
            \underset{k \in s_{M}}{\sum}\;d_{k}\,w_{k}\,Y_{k}
            \;\right\vert
            \begin{array}{c}
                s,s_{R},
                \\
                \Zdot\,,\Xobs
                \end{array}
            \!\right],
        \;\;
        \textnormal{by Lemma \ref{PopTotalImputationError}(ii)}
    \\
    & = &
        \left(\,\dfrac{1}{ \overset{{\color{white}.}}{\widehat{T}(s\,;\zdot)} }\,\right)^{\!2}
        \cdot
        \Var_{Y}\!\left[\;
            \left.
            \underset{l \in s_{R}}{\sum}\,W^{(+)}_{l}\,Y_{l}
            \,-\,
            \underset{k \in s_{M}}{\sum}\;d_{k}\,w_{k}\,Y_{k}
            \;\right\vert
            \begin{array}{c}

                \end{array}
            \!\right]
        % \;\;

    \\
    & = &
        \left(\,\dfrac{1}{ \overset{{\color{white}.}}{\widehat{T}(s\,;\zdot)} }\,\right)^{\!2}
        \cdot
        \left(\;
            \underset{l \in s_{R}}{\sum}
            \!\left(W^{(+)}_{l}\right)^{2}

    \\

                Y_{l}

                \Zdot\,,\Xobs
                \right.
                \right]

        \left(\;
            \underset{l \in s_{R}}{\sum}
                \left(W^{(+)}_{l}\right)^{2}\sigma_{l}^{2}

            \dfrac{
                1
                }{
                \overset{{\color{white}.}}{
                    \widehat{T}(s\,;\Zdot)
                    }
                }
            \cdot
            \left(\,
                \widetilde{T}_{Y}
                \, - \,
                \widehat{T}_{Y}
                \,\right)
            \;\right\vert
            \begin{array}{c}

                \\
                \Zdot\,,\Xobs
                \end{array}
            \!\right]
    \;\; = \;\;
        \dfrac{
            1
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)
                }
            }
        \cdot
        E_{Y}\!\left[\;\,
            \left.
            \widetilde{T}_{Y}
            \, - \,
            \widehat{T}_{Y}
            \;\right\vert\!
            \begin{array}{c}
                s,s_{R},
                \\
                \Zdot\,,\Xobs
                \end{array}
            \!\right]
    \\
    & = &
        \dfrac{
            1
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)
                }
            }
        \cdot
        E_{Y}\!\left[\;
            \left.
            W^{(+)}_{0}
            \,+\,
            \underset{l \in s_{R}}{\sum}\,W^{(+)}_{l}\,Y_{l}
            \,-\,
            \underset{k \in s_{M}}{\sum}\;d_{k}\,w_{k}\,Y_{k}
            \;\,\right\vert\!
            \begin{array}{c}
                s,s_{R},
                \\
                \Zdot\,,\Xobs
                \end{array}
            \!\right],
        \quad
        \textnormal{by Lemma \ref{PopTotalImputationError}(ii)}
    \\
    & = &
        \dfrac{
            1
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)
                }
            }
        \cdot
        \left(\;
            W^{(+)}_{0}
            \;+\;
            \underset{l \in s_{R}}{\sum}\;
            W^{(+)}_{l}

                \begin{array}{c}

                \!\right]
            \;\right)
    \\
    & = &
        \dfrac{
            1
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)
                }
            }
        \cdot
        \left(\;
            W^{(+)}_{0}
            \;+\;
            \underset{l \in s_{R}}{\sum}\;
            W^{(+)}_{l}
            \cdot
            E_{Y}\!\left[\;
                Y_{l}
                \,\left\vert\,
                \Zdot\,,\Xobs
                \right.
                \right]
            \,\;-\;
            \underset{k \in s_{M}}{\sum}\;
            d_{k}\,w_{k}

            E_{Y}\!\left[\;Y_{k}
                \,\left\vert\,
                \Zdot\,,\Xobs
                \right.
                \,\right]

    &&
        \quad
        \textnormal{since \,$\Ydot \perp (\Sdot,\Rdot) \;\;\vert\;\, \Zdot\,,\Xobs$}
    \\
    & = &
        \dfrac{
            1
            }{
            \overset{{\color{white}.}}{
                \widehat{T}(s\,;\zdot)
                }
            }
        \cdot
        \left(\;
            W^{(+)}_{0}
            \;+\;
            \underset{l \in s_{R}}{\sum}\;W^{(+)}_{l}\,\mu_{l}
            \,\;-\;
            \underset{k \in s_{M}}{\sum}\,
                d_{k}\,w_{k}\,\mu_{k}

    \\
    &&
        \quad
        \textnormal{by definition of
            \,$\mu_{k} \;:=\, E_{Y}\!\left[\;Y_{k}
                \,\left\vert\,
                \Zdot\,,\Xobs
                \right.
                \,\right]$}
    \end{eqnarray*}
    This completes the proof of Claim 3.

    \noindent
    The required result now follows immediately from Claim 1, Claim 2, and Claim 3.
    \qed

\item

    &&

            (\,\widetilde{R}_{Y/Z} - \widehat{R}_{Y/Z}\,)

\proof
\begin{enumerate}
\item
    Nothing to prove.
\item
    % \begin{eqnarray*}
    % &&
    %     E_{Y}\!\left[\;
    %         \left(\,\widetilde{R}_{Y/Z} \, - \, \widehat{R}_{Y/Z}\,\right)

    %         \left(\,  \widehat{R}_{Y/Z} \, - \, R_{Y/Z}\,\right)

    %             \,\right)
    %         \!
    %         \left(\,
    %             \dfrac{ \widehat{T}_{Y} }{ \overset{{\color{white}.}}{\widehat{T}(s\,;\Zdot)} }

    The required result follows from the following general fact:
    \begin{equation*}
    \Cov(\,X,Y\,)
    \;\; := \;\;
        E\!\left[\,\overset{{\color{white}.}}{(X - \mu_{X})(Y - \mu_{Y})}\,\right]
    \;\; = \;\;
        \cdots
    \;\; = \;\;
        E\!\left[\,\overset{{\color{white}.}}{XY}\,\right]
        \; - \;
        \mu_{X}\,\mu_{Y}\,,
    \end{equation*}
    which implies
    \begin{equation*}
    E\!\left[\,\overset{{\color{white}.}}{XY}\,\right]
    \;\; = \;\;
        \Cov(\,X,Y\,)
        \; + \;
        \mu_{X}\,\mu_{Y}
    \end{equation*}
\item
    We compute:
    \begin{eqnarray*}
    &&
        \Cov_{Y}\!\left(\,
            \left.
            \widetilde{T}_{Y} - \widehat{T}_{Y}
            \,,\,
            \widehat{T}_{Y} - T_{Y}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right)
    \\
    & = &
        \Cov_{Y}\!\left(\,
            \left.
            W^{(+)}_{0}
            \,+\,
            \underset{l \in s_{R}}{\sum}\,W^{(+)}_{l}\,Y_{l}
            \;-
            \underset{k \in s_{M}}{\sum}\,d_{k}\,w_{k}\,Y_{k}
            \;,\;
            \underset{k \in s}{\sum}\;d_{k}w_{k}Y_{k}
            \;-
            \underset{k \in U}{\sum}\;d_{k}Y_{k}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right),
            \quad
            \textnormal{by Lemma \ref{PopTotalImputationError}(ii)}
    \\
    & = &
        \Cov_{Y}\!\left(\;
            \left.
            \underset{l \in s_{R}}{\sum}\,W^{(+)}_{l}\,Y_{l}
            \;-
            \underset{k \in s_{M}}{\sum}\,d_{k}\,w_{k}\,Y_{k}
            \;,\;
            \underset{k \in s}{\sum}\;d_{k}(w_{k}-1)Y_{k}
            \;-
            \underset{k \in U \backslash s}{\sum}d_{k}Y_{k}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right)
    \\
    & = &
        \underset{l \in s_{R}}{\sum}\,
        d_{l}\,W^{(+)}_{l}\,(w_{l}-1)\,\Var_{Y}\!\left[\,
            Y_{l}
            \left\vert\,
            s,s_{R},\Xobs
            \right.
            \right]
        \;-\;
        \underset{k \in s_{M}}{\sum}\,

            Y_{k}
            \left\vert\,
            s,s_{R},\Xobs
            \right.
            \right],
    \\
    &&
        \quad\quad\quad
        \textnormal{since
            \,$
            \Cov\!\left(\,Y_{a},Y_{b}\,\left\vert\,s,s_{R},\Xobs\right.\right)

            \Cov\!\left(\,Y_{a},Y_{b}\,\left\vert\,\Xobs\right.\right)

            0
            $,\, whenever \,$a \neq b$}
    \\
    & = &
        \underset{l \in s_{R}}{\sum}\,
        d_{l}\,W^{(+)}_{l}\,(w_{l}-1)\,\Var_{Y}\!\left[\,
            Y_{l}
            \left\vert\,
            \Xobs
            \right.
            \right]
        \,-\,
        \underset{k \in s_{M}}{\sum}\,
        d_{k}\,w_{k}\,(w_{k}-1)\,\Var_{Y}\!\left[\,
            Y_{k}^{2}
            \left\vert\,
            \Xobs
            \right.
            \right],
        \;\;
        \textnormal{since $\Ydot \perp (\Sdot,\Rdot) \;\vert\; \Xobs$}
    \\
    & = &

        d_{l}\,W^{(+)}_{l}\,(w_{l}-1)\,\sigma_{l}^{2}
        \;\;-\;
        \underset{k \in s_{M}}{\sum}\,
        d_{k}\,w_{k}\,(w_{k}-1)\,\sigma_{k}^{2}
    \end{eqnarray*}
    % where

    % while
    % the fourth equality follows from the conditional independence
    % \,$\Ydot \perp (\Sdot,\Rdot) \;\vert\; \Xobs$.\,
    \vskip 0.1cm
    \noindent
    This completes the proof of the Lemma.
    \qed
\end{enumerate}

    \left(\;

        \underset{k \in s_{M}^{(j)}}{\sum}\,d_{k}\,w_{k}\,\varphi^{(j)}_{{\color{red}l}k}
        \,\right),
\;\;
\textnormal{for each \,$l \in s_{R}$}
\end{equation*}
\vskip 0.2cm
\noindent
Suppose the quantities
\,$
\Sdot\,,\,
\Rdot\,,\,
\Ydot\,,\,
\Xdotdot\,,\,
\dbullet
$\,
satisfy the probabilistic assumptions specified in Subsection \ref{ProbabilisticFramework}.
\vskip 0.2cm
\noindent
Then, the following summarizes the results in \cite{Beaumont2011}:
\begin{itemize}
\item
    By Lemma \ref{DecompositionMSETtilde}, we have the following decomposition:
    \begin{eqnarray*}
    \MSE\!\left(\;
        \left.
        \widetilde{T}_{Y}
        \;\right\vert\,
        \Xobs
        \,\right)
    & := &
        E\!\left[\,
            \left.
            (\,\widetilde{T}_{Y} - T_{Y}\,)^{2}
            \,\right\vert\,
            \Xobs
            \,\right]
    \\
    & = &
        E_{Y}\!\left[\,
        \Var_{SR}\!\left(\,
            \left.
            \widehat{T}_{Y}
            \,\right\vert\,
            \Ydot,
            \Xobs
            \,\right)
            \,\right]
        \;+\;
        E_{SR}\!\left[\,
        E_{Y}\!\left[\,
            \left.
            (\,\widetilde{T}_{Y} - \widehat{T}_{Y}\,)^{2}
            \,\right\vert\,
            s,s_{R},\Xobs
            \,\right]
            \,\right]
    \\
    &&
        +\;2\,
        E_{SR}\!\left[\,
        E_{Y}\!\left[\,
            \left.
            (\,\widetilde{T}_{Y} \,-\, \widehat{T}_{Y}\,)(\,\widehat{T}_{Y} \,-\, T_{Y}\,)
            \,\right\vert\,
            s,s_{R},\Xobs
            \,\right]
            \,\right]
    \end{eqnarray*}
    We use the following nomenclature to refer the three summands in the right-hand side:
    \begin{eqnarray*}
    \left(\begin{array}{c}
        \textnormal{sampling}
        \\
        \textnormal{\;\,variance component\;\,}
        \end{array}\right)
    & := &
        E_{Y}\!\left[\,
        \Var_{SR}\!\left(\,
            \left.
            \widehat{T}_{Y}
            \,\right\vert\,
            \Ydot,
            \Xobs
            \,\right)
            \,\right]
    \\
    \left(\begin{array}{c}
        \textnormal{nonresponse}
        \\
        \textnormal{\;\,variance component\;\,}
        \end{array}\right)
    & := &
        E_{SR}\!\left[\,
        E_{Y}\!\left[\,
            \left.
            (\,\widetilde{T}_{Y} - \widehat{T}_{Y}\,)^{2}
            \,\right\vert\,
            s,s_{R},\Xobs
            \,\right]
            \,\right]
    \\
    \left(\begin{array}{c}
        \textnormal{sampling-nonresponse}
        \\
        \textnormal{covariance}
        \end{array}\right)
    & := &
        2\,
        E_{SR}\!\left[\,
        E_{Y}\!\left[\,
            \left.
            (\,\widetilde{T}_{Y} \,-\, \widehat{T}_{Y}\,)(\,\widehat{T}_{Y} \,-\, T_{Y}\,)
            \,\right\vert\,
            s,s_{R},\Xobs
            \,\right]
            \,\right]
    \end{eqnarray*}
\item
    By Lemma \ref{SamplingVariance}, the sampling variance component can be estimated as follows:
    \begin{eqnarray*}
    \left(\!\begin{array}{c}
        \textnormal{sampling}
        \\
        \textnormal{variance component}
        \end{array}\!\right)
    & := &
        E_{Y}\!\left[\,
        \Var_{SR}\!\left(\,
            \left.
            \widehat{T}_{Y}
            \,\right\vert\,
            \Ydot,
            \Xobs
            \,\right)
            \,\right]
    \;\; = \;\;
        E_{SR}\,
        E_{Y_{R}}\,
        E_{Y_{M}}\!\left[\,
            v(\,s\,;\Ydot\,)
            \,\left\vert\;
            s,s_{R},\YsubR,\Xobs
            \right.
            \,\right]
    \\
    & {\color{red}\approx} &
    E_{Y_{M}}\!\left[\,
        v(\,s\,;\Ydot\,)
        \left\vert\,
        s,s_{R},\YsubR=y_{R},\Xobs
        \right.
        \right]
    \;\; = \;\;
        {\color{blue}
        v(\,s\,;\widetilde{y}_{\bullet}\,)
        \;+\;
        \underset{k \in s_{M}}{\sum}\;
        d_{k}
        \left(1-\dfrac{1}{w_{k}}\right)
        w_{k}^{2}\,\sigma_{k}^{2}
        }\,,
    \end{eqnarray*}
    where
    \,$v(\,s\,;y_{\bullet}\,)$\,
    is the (design-unbiased) Horvitz-Thompson estimator for sampling variance, i.e.,
    \begin{equation*}
    v(\,s\,;y_{\bullet}\,)
    \;\; := \;\;
        \underset{k \in s}{\sum}\,
        \underset{l \in s}{\sum}\,
        \dfrac{\pi_{kl} - \pi_{k}\pi_{l}}{\pi_{kl}}\,
        (d_{k}w_{k}y_{k})\,(d_{l}w_{l}y_{l})\,,
    \end{equation*}
    and
    \begin{equation*}
    \widetilde{y}_{k}
    \;\; := \;\;
        \left\{\begin{array}{cl}
            y_{k}, & \textnormal{for \,$k \,\in\, s_{R}$}
            \\
            % \overset{{\color{white}1}}{E_{Y}\!\left[\;Y_{k}\,\left\vert\;\Xobs\right.\right]}, & \textnormal{for \,$k \,\in\, s_{M} \,:=\, s \,\backslash\, s_{R}$}
            \overset{{\color{white}\textnormal{\large1}}}{\mu_{k}}, & \textnormal{for \,$k \,\in\, s_{M} \,:=\, s \,\backslash\, s_{R}$}
            \end{array}\right.
    \end{equation*}
\item
    By Lemma \ref{NonresponseVariance}, the nonresponse variance component can be estimated as follows:
    \begin{eqnarray*}
    \left(\!\begin{array}{c}
        \textnormal{nonresponse}
        \\
        \textnormal{variance}
        \\
        \textnormal{{\color{white}..}component{\color{white}..}}
        \end{array}\!\right)
    & := &
        E_{SR}\!\left[\,
        E_{Y}\!\left[\,
            \left.
            (\,\widetilde{T}_{Y} - \widehat{T}_{Y}\,)^{2}
            \,\right\vert\,
            s,s_{R},\Xobs
            \,\right]
            \,\right]
    \;\; {\color{red}\approx} \;\;
        E_{Y}\!\left[\,
            \left.
            (\,\widetilde{T}_{Y} - \widehat{T}_{Y}\,)^{2}
            \,\right\vert\,
            s,s_{R},\Xobs
            \,\right]
    \\
    & = &
        {\color{blue}
        \left(\;
            \underset{l \in s_{R}}{\sum}
                \left(W^{(+)}_{l}\right)^{2}\sigma_{l}^{2}
            \,+\,
            \underset{k \in s_{M}}{\sum}\,
                d_{k}\,w_{k}^{2}\,\sigma_{k}^{2}
            \,\right)
        \;+\;
        \left(\,
            W^{(+)}_{0}
            \,+\,
            \underset{l \in s_{R}}{\sum}\;W^{(+)}_{l}\,\mu_{l}
            \,-\,
            \underset{k \in s_{M}}{\sum}\;d_{k}\,w_{k}\,\mu_{k}
            \,\right)^{2}
        }
    \end{eqnarray*}
\item
    By Lemma \ref{MixedVariance}, the mixed variance component can be estimated as follows:
    \begin{eqnarray*}
    \left(\!\begin{array}{c}
        \textnormal{sampling-nonresponse}
        \\
        \textnormal{covariance}
        \end{array}\!\right)
    & := &
        2\,
        E_{SR}\!\left[\,
        E_{Y}\!\left[\,
            \left.
            (\,\widetilde{T}_{Y} \,-\, \widehat{T}_{Y}\,)(\,\widehat{T}_{Y} \,-\, T_{Y}\,)
            \,\right\vert\,
            s,s_{R},\Xobs
            \,\right]
            \,\right]
    \\
    & \approx &
        2\,
        E_{Y}\!\left[\,
            \left.
            (\,\widetilde{T}_{Y} \,-\, \widehat{T}_{Y}\,)(\,\widehat{T}_{Y} \,-\, T_{Y}\,)
            \,\right\vert\,
            s,s_{R},\Xobs
            \,\right]
    \\
    & = &
        2\cdot
        \Cov_{Y}\!\left(\,
            \left.
            \widetilde{T}_{Y} - \widehat{T}_{Y}
            \,,\,
            \widehat{T}_{Y} - T_{Y}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right)
    \\
    &&
        +\;
        2\cdot
        E_{Y}\!\left[\;
            \left.
            \widetilde{T}_{Y} - \widehat{T}_{Y}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right]
        \cdot
        {\color{red}E_{Y}\!\left[\;
            \left.
            \widehat{T}_{Y} - T_{Y}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right]}
    \\
    & {\color{red}\approx} &
        2\cdot
        \Cov_{Y}\!\left(\,
            \left.
            \widetilde{T}_{Y} - \widehat{T}_{Y}
            \,,\,
            \widehat{T}_{Y} - T_{Y}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right)
    \\
    & = &
        {\color{blue}
        2\,\cdot\!
        \underset{l \in s_{R}}{\sum}\,
            d_{l}\,W^{(+)}_{l}\,(w_{l}-1)\,\sigma_{l}^{2}
        \;\;-\;
        2\,\cdot\!
        \underset{k \in s_{M}}{\sum}\,
            d_{k}\,w_{k}\,(w_{1}-1)\,\sigma_{k}^{2}
        }
    \end{eqnarray*}
\item
    \textbf{Important Remark:}
    \vskip 0.05cm
    \noindent
    Note that the above estimation/approximation of the mixed variance component
    disregards the following term:
    \begin{equation*}
    2\,\cdot E_{Y}\!\left[\;
        \left.
        \widetilde{T}_{Y} - \widehat{T}_{Y}
        \;\right\vert\,
        s,s_{R},\Xobs
        \,\right]
    \cdot
    {\color{red}E_{Y}\!\left[\;
        \left.
        \widehat{T}_{Y} - T_{Y}
        \;\right\vert\,
        s,s_{R},\Xobs
        \,\right]}
    \end{equation*}
    The reason this has to be done in practice is that, in practice,
    there is usually insufficient data/information to compute or estimate
    the quantity
    \,$
    E_{Y}\!\left[\;
        \left.
        \widehat{T}_{Y} - T_{Y}
        \;\right\vert\,
        s,s_{R},\Xobs
        \,\right]
    $\,
    (indicated in red above), as \,$T_{Y}$\, involves population units
    that are not selected in the sample.
    \vskip 0.1cm
    \noindent
    Obviously, the above estimation/approximation of the mixed variance component
    is acceptable only if the quantity
    \,$
    2\,\cdot E_{Y}\!\left[\;
        \left.
        \widetilde{T}_{Y} - \widehat{T}_{Y}
        \;\right\vert\,
        s,s_{R},\Xobs
        \,\right]
    \cdot
    {\color{black}E_{Y}\!\left[\;
        \left.
        \widehat{T}_{Y} - T_{Y}
        \;\right\vert\,
        s,s_{R},\Xobs
        \,\right]}
    $\,
    is small.
    \vskip 0.1cm
    \noindent
    For this, in practice, we thus {\color{red}tacitly assume} that the quantity
    \,$
    E_{Y}\!\left[\;
        \left.
        \widetilde{T}_{Y} \overset{{\color{white}.}}{-} \widehat{T}_{Y}
        \;\right\vert\,
        s,s_{R},\Xobs
        \,\right]
    $\,
    is small;
    recall that this quantity is called the \textbf{conditional model bias}
    (see Lemma \ref{biasTtilde}), and
    it can be expressed as follows:
    \begin{eqnarray*}
    E_{Y}\!\left[\;
        \left.
        \widetilde{T}_{Y} \overset{{\color{white}.}}{-} \widehat{T}_{Y}
        \;\right\vert\,
        s,s_{R},\Xobs
        \,\right]
    & = &
        \underset{k\in{\color{red}s_{M}}}{\sum}\,d_{k}\,w_{k}\,
        E_{Y}\!\left[\;
            \left.
            Y^{*}_{k} \overset{{\color{white}.}}{-} Y_{k}
            \;\right\vert\,
            s,s_{R},\Xobs
            \,\right]
    \end{eqnarray*}
    This harks back to our earlier remark (Remark \ref{remarkSmallUnitBias}) that,
    in practice, we would like to choose an imputation strategy such that
    \begin{equation*}
    E_{Y}\!\left[\;
        \left.
        Y^{*}_{k} \overset{{\color{white}.}}{-} Y_{k}
        \;\right\vert\,
        s,s_{R},\Xobs
        \,\right]
    \;\; \approx \;\;
        0,
    \quad
    \textnormal{for each \,$k \in s_{M}$}
    \end{equation*}
    See the paragraph after Equation (5.2), p.175, \cite{Beaumont2011}.
\end{itemize}
