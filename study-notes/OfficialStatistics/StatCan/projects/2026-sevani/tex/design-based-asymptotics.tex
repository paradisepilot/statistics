\begin{definition}[Stochastic order notations]
\mbox{}
\vskip 0.1cm
\noindent
Suppose:
\begin{itemize}
\item
    $\left(\,\Omega,\mathscr{A},\mu\,\right)$\,
    is a probability space,
\item
    $X_{1}, X_{2}, \ldots : \Omega \longrightarrow \Re^{p}$
    form a sequence of $\Re^{p}$-valued random variables, and
\item
    $Y_{1}, Y_{2}, \ldots : \Omega \longrightarrow \Re$
    form a sequence of $\Re$-valued random variables.
\end{itemize}
Then, we define the following \textbf{stochastic order notations}:
\begin{enumerate}
\item
    $X_{n} \,=\,O(Y_{n})$\, almost surely if
    \begin{equation*}
    P\!\left(\,X_{n} = O(Y_{n})\,\right) \;=\; 1\,;
    \quad\quad\textnormal{equivalently,}\quad\quad
    \mu\!\left(\,\left\{\;\,
        \omega \overset{{\color{white}.}}{\in} \Omega
        \;\;\left\vert\;
        \begin{array}{c}
        \exists\;\, C_{\omega} > 0 \;\,\textnormal{such that}
        \\
        \Vert\,X_{n}(\omega)\,\Vert \leq C_{\omega}\vert\,Y_{n}(\omega))\,\vert,
        \\
        \textnormal{for each \,$n \in \N$}
        \end{array}
        \right.
        \;\right\}\,\right) \;=\; 1
    \end{equation*}
\item
    $X_{n} \,=\,o(Y_{n})$\, almost surely if
    \begin{equation*}
    P\!\left(\,\underset{n\rightarrow\infty}{\lim}\,\dfrac{X_{n}}{Y_{n}} = 0\,\right) \;=\; 1\,;
    \quad\quad\textnormal{equivalently,}\quad\quad
    \mu\!\left(\,\left\{\;\,
        \left.
        \omega \overset{{\color{white}1}}{\in} \Omega
        \;\;\right\vert\;
        \underset{n\rightarrow\infty}{\lim}\,\dfrac{X_{n}(\omega)}{Y_{n}(\omega)} = 0
        \;\right\}\,\right) \;=\; 1
    \end{equation*}
\item
    $X_{n} \,=\,O_{p}(Y_{n})$\, if, for each \,$\varepsilon > 0$\,,
    there exists \,$C_{\epsilon} > 0$\, such that    
    \begin{equation*}
    P\!\left(\,
        \Vert\,X_{n}\,\Vert
        \,\overset{{\color{white}1}}{<}\,
        C_{\varepsilon}\vert\,Y_{n}\,\vert
        \,\right)
    \;\; = \;\;
    \mu\!\left(\,\left\{\;\,
        \omega \in \Omega
        \,\;\left\vert\;\;
        \Vert\,X_{n}(\omega)\,\Vert
        \,\overset{{\color{white}1}}{<}\,
        C_{\varepsilon}\vert\,Y_{n}(\omega)\,\vert
        \right.
        \;\right\}\,\right)
    \;\; > \;\;
        1 - \varepsilon\,,
    \quad
    \textnormal{for each \,$n = 1, 2, \ldots$}
    \end{equation*}
\item
    $X_{n} \,=\,o_{p}(Y_{n})$\, if
    \,$X_{n}/Y_{n}$\,
    converges to \,$0$\, in probability, i.e.,
    for each \,$\varepsilon > 0$,\,
    \begin{equation*}
    \underset{n\rightarrow\infty}{\lim}\;
    P\!\left(\;
        \dfrac{\Vert\,X_{n}\,\Vert}{\vert\,Y_{n}\,\vert}
        \,<\,
        \varepsilon
        \;\right)
    \;\; = \;\;
    \underset{n\rightarrow\infty}{\lim}\;
    \mu\!\left(\,\left\{\;\;
        \omega \in \Omega
        \,\;\left\vert\,\;
        \dfrac{\Vert\,X_{n}(\omega)\,\Vert}{\vert\,Y_{n}(\omega)\,\vert}
        \,<\,
        \varepsilon
        \right.
        \;\right\}\,\right)
    \;\; = \;\;
        1
    \end{equation*}
\end{enumerate}
\end{definition}

\subsection{Some asymptotics of the Horvitz-Thompson estimator for population totals in terms of stochastic order notations}

          %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

\vskip 0.5cm
\begin{theorem}\label{AsymptoticsHTForDomainTotal}
\mbox{}
\vskip 0.05cm
\noindent
\textbf{Setting}
\vskip 0.1cm
\noindent
Suppose:
\begin{itemize}
\item
    $\left\{\;\overset{{\color{white}.}}{U_{\nu}}\;\right\}_{\nu=1}^{\infty}$\,
    is a sequence of (nonempty) finite populations whose sequence
    $\left\{\,N_{\nu} \overset{{\color{white}-}}{:=} \vert\,U_{\nu}\,\vert\,\right\}_{\nu=1}^{\infty}$\,
    of population sizes tends to infinity as $\nu \rightarrow \infty$, i.e.,
    \,$\underset{\nu\rightarrow\infty}{\lim}\,N_{\nu} \,=\, \infty$.\,
\item
    For each \,$\nu \in \N$,\,
    \,$p_{\nu} : \mathcal{P}(U_{\nu}) \longrightarrow [\,0,1)$\,
    is a {\color{red}fixed-sample-size} sampling design on \,$U_{\nu}$\,
    such that
    \begin{equation*}
    \pi^{(\nu)}_{i}
    \;\; := \;\;
        \underset{s\,\ni\,i}{\sum}\;\,
        p_{\nu}\!\left(\,s\,\right)
    \;\; > \;\;
        0\,,
    \quad
    \textnormal{for each \,$i \in U_{\nu}$,\, and for each \,$\nu\in\N$}
    \end{equation*}
\item
    For each \,$\nu \in \N$,\,
    \,$y^{(\nu)} : U_{\nu} \longrightarrow \Re$\,
    is an $\Re$-valued population characteristic defined on \,$U_{\nu}$.\,
\end{itemize}
\vskip 0.2cm
\noindent
\textbf{Notations}
\begin{itemize}
\item
    For each \,$\nu\in\N$,\,
    \,$n_{\nu}$\, is the fixed sample size of the sampling design \,$p_{\nu}$, i.e.,\,
    \begin{equation*}
    n_{\nu}
    \;\; := \;\;
        \underset{k\,\in\,U_{\nu}}{\sum}\,I^{(\nu)}_{i}(s)\,,
    \quad
    \textnormal{for each sample \,$s \subset U_{\nu}$\, with \,$p_{\nu}(s) > 0$}
    \end{equation*}
    Note that left-hand side \,$n_{\nu}$\, is indeed well-defined
    since the right-hand side is independent of the sample \,$s$ (as long as $p_{\nu}(s) > 0$),\,
    by the fixed-sample-size assumption on the sampling design \,$p_{\nu}$.
\item
    For each \,$\nu\in\N$,\,
    \,$Y_{\nu}$\, is the population total of \,$y^{(\nu)}$,\, i.e.,
    \;$
    Y_{\nu}
    \;\, := \,
        \underset{i \in U_{\nu}}{\sum}\,y^{(\nu)}_{i}
    $.\,
\item
    For each \,$\nu\in\N$,\,
    \,$\widehat{Y}^{\textnormal{(HT)}}_{\nu}$\,
    is the Horvitz-Thompson estimator for the population total \,$Y_{\nu}$,\, i.e.,
    \begin{equation*}
    \widehat{Y}^{\textnormal{(HT)}}_{\nu}(\,s\,)
    \;\; := \;\;
        \underset{i \in s}{\sum}\;
        \dfrac{ y^{(\nu)}_{i} }{ \pi^{(\nu)}_{i} }
    \end{equation*}
\end{itemize}
\vskip 0.2cm
\noindent
\textbf{Hypotheses (``regularity conditions'')}
\vskip 0.1cm
\noindent
Suppose the entities above satisfy the following hypotheses (``regularity conditions''):
\begin{itemize}
\item
    Design weight condition:
    There exists \,$\lambda > 0$\,
    such that
    \begin{equation*}
    \dfrac{1}{\pi^{(\nu)}_{i}}
    \;\leq\;
        \dfrac{1}{\lambda}
        \cdot
        \dfrac{N_{\nu}}{n_{\nu}}\,,
    \quad
    \textnormal{for each \,$i \in U_{\nu}$,\; and for each \,$\nu \in \N$}
    \end{equation*}
\item
    Sampling covariance condition:
    There exists a constant \,$C > 0$\, such that,
    for all \,$i \neq j$,\,
    \begin{equation*}
    \left\vert\; \pi^{(\nu)}_{ij} - \pi^{(\nu)}_{i}\,\pi^{(\nu)}_{j} \;\right\vert
    \;\; \leq \;\;
        C\cdot\dfrac{n_{\nu}}{N_{\nu}^{2}}\,,
    \quad
    \textnormal{for each \,$i,j \in U_{\nu}$ with $i \neq j$,\, for each \,$\nu\in\N$}
    \end{equation*}
\item
    Moment condition:\,
    There exists \,$M > 0$\, such that
    \begin{equation*}
    \dfrac{1}{N_{\nu}}\cdot\underset{i\,\in\,U_{\nu}}{\sum}\!\left(\,y^{(\nu)}_{i}\,\right)^{2}
    \;\; \leq \;\; M \;\; < \;\; \infty\,,
    \quad
    \textnormal{for each \,$\nu\in\N$}
    \end{equation*}
\end{itemize}
\vskip 0.2cm
\noindent
\textbf{Conclusions}
\vskip 0.1cm
\noindent
Then, we have:
\begin{equation*}
\dfrac{
    {\color{white}.}
    \widehat{Y}^{\textnormal{(HT)}}_{\nu} \,-\, Y_{\nu}
    {\color{white}.}
    }{
    N_{\nu}
    }
\;\; = \;\;
    O_{p}\!\left(\,\dfrac{1}{\sqrt{n_{\nu}}}\,\right)
\end{equation*}
equivalently, but more explicitly,
for each \,$\varepsilon > 0$,\, there exists \,$C_{\varepsilon} > 0$\, such that
\begin{equation*}
P\!\left(\,
    \dfrac{
        {\color{white}.}
        \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu} \,-\, Y_{\nu}\,\right\vert
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{N_{\nu}}
        }
    \,<\,
    \dfrac{C_{\varepsilon}}{\sqrt{n_{\nu}}}
    \;\right)
\;\; = \;\;
\mu_{\nu}\!\left(\left\{\;\;
    s \in \mathcal{P}(U_{\nu})
    \;\;\left\vert\;\;
    \dfrac{
        {\color{white}.}
        \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}(s) \,-\, Y_{\nu}\,\right\vert
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{N_{\nu}}
        }
    \,<\,
    \dfrac{C_{\varepsilon}}{\sqrt{n_{\nu}}}
    \right.
    \;\right\}\right)
\;\; > \;\;
    1 \,-\, \varepsilon\,,
\quad
\textnormal{for each \,$\nu \in \N$}
\end{equation*}
\end{theorem}

\vskip 0.5cm
\proof
\vskip 0.1cm
\noindent
\textbf{Claim 1:}\quad
$
\Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)
\,=\,
O\!\left(\,\dfrac{N_{\nu}^{2}}{n_{\nu}}\,\right)
$
\vskip 0.1cm
\noindent
Proof of Claim 1:\;\;
Recall that
\begin{eqnarray*}
\Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)
& = &
    \underset{k \in U}{\sum}\;
    \underset{l \in U}{\sum}
    \left(\,
        \pi^{(\nu)}_{kl}
        \, - \,
        \pi^{(\nu)}_{k}\pi^{(\nu)}_{l}
        \,\right)
    \dfrac{y^{(\nu)}_{k}}{\pi^{(\nu)}_{k}}
    \,
    \dfrac{y^{(\nu)}_{l}}{\pi^{(\nu)}_{l}}
\\
& = &
    \underset{k \in U}{\sum}
    \left(\,
        \dfrac{
            1 \,-\, \pi^{(\nu)}_{k}
            }{
            \pi^{(\nu)}_{k}
            }
        \,\right)
    \left(y^{(\nu)}_{k}\right)^{2}
    \; + \;
    \underset{k\,\neq\,l}{\underset{k,l\,\in\,U}{\sum\sum}}
    \left(\,
        \pi^{(\nu)}_{kl}
        \, - \,
        \pi^{(\nu)}_{k}\pi^{(\nu)}_{l}
        \,\right)
    \dfrac{y^{(\nu)}_{k}}{\pi^{(\nu)}_{k}}
    \,
    \dfrac{y^{(\nu)}_{l}}{\pi^{(\nu)}_{l}}
\end{eqnarray*}
Now, observe:
\begin{eqnarray*}
\left(\,\begin{array}{c}
    \textnormal{first summand of}
    \\
    \Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)
    \end{array}\,\right)
& = &
    \underset{k \in U}{\sum}
    \left(\,
        \dfrac{
            1 \,-\, \pi^{(\nu)}_{k}
            }{
            \pi^{(\nu)}_{k}
            }
        \,\right)
    \left(y^{(\nu)}_{k}\right)^{2}
\;\; \leq \;\;
    \underset{k \in U}{\sum}
    \left(\,
        \dfrac{ 1 }{ \pi^{(\nu)}_{k} }
        \,\right)
    \left(y^{(\nu)}_{k}\right)^{2}
\\
& \leq &
    \underset{k \in U}{\sum}
    \left(\,
        \dfrac{1}{\lambda} \cdot \dfrac{ N_{\nu} }{ n_{\nu} }
        \,\right)
    \left(y^{(\nu)}_{k}\right)^{2}\,,
% \\
% &&
    \quad
    \textnormal{by the Design Weight regularity condition}
\\
& \leq &
    \left(\,
        \dfrac{1}{\lambda} \cdot \dfrac{ N_{\nu} }{ n_{\nu} }
        \,\right)
    \cdot
    \underset{k \in U}{\sum}
    \left(y^{(\nu)}_{k}\right)^{2}
\;\; = \;\;
    \left(\,
        \dfrac{1}{\lambda} \cdot \dfrac{ N_{\nu}^{2} }{ n_{\nu} }
        \,\right)
    \left(\,
        \dfrac{1}{N_{\nu}}
            \cdot
        \underset{k \in U}{\sum}
        \left(y^{(\nu)}_{k}\right)^{2}
        \,\right)
\\
& \leq &
    \left(\,
        \dfrac{1}{\lambda} \cdot \dfrac{ N_{\nu}^{2} }{ n_{\nu} }
        \,\right)
    \cdot
    M\,,
    \quad
    \textnormal{by the Moment regularity condition},
\end{eqnarray*}
from which it follows that
\begin{equation*}
\left(\,\begin{array}{c}
    \textnormal{first summand of}
    \\
    \Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)
    \end{array}\,\right)
\;\; = \;\;
    \underset{k \in U}{\sum}
    \left(\,
        \dfrac{
            1 \,-\, \pi^{(\nu)}_{k}
            }{
            \pi^{(\nu)}_{k}
            }
        \,\right)
    \left(y^{(\nu)}_{k}\right)^{2}
\;\; = \;\;
    O\!\left(\,\dfrac{N_{\nu}^{2}}{n_{\nu}}\,\right)
\end{equation*}
Next, observe:
\begin{eqnarray*}
\left\vert\,\begin{array}{c}
    \textnormal{second summand of}
    \\
    \Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)
    \end{array}\,\right\vert
& \leq &
    \underset{k\,\neq\,l}{\underset{k,l\,\in\,U}{\sum\sum}}\;
    \left\vert\,
        \pi^{(\nu)}_{kl}
        \, - \,
        \pi^{(\nu)}_{k}\pi^{(\nu)}_{l}
        \,\right\vert
    \cdot
    \dfrac{\left\vert\,y^{(\nu)}_{k}\,\right\vert}{\pi^{(\nu)}_{k}}
    \cdot
    \dfrac{\left\vert\,y^{(\nu)}_{l}\,\right\vert}{\pi^{(\nu)}_{l}}
% \\
% & \leq &
%     \underset{k\,\neq\,l}{\underset{k,l\,\in\,U}{\sum\sum}}\;
%     \dfrac{C \cdot n_{\nu}}{N_{\nu}^{2}}
%     \cdot
%     \dfrac{\left\vert\,y^{(\nu)}_{k}\,\right\vert}{\pi^{(\nu)}_{k}}
%     \cdot
%     \dfrac{\left\vert\,y^{(\nu)}_{l}\,\right\vert}{\pi^{(\nu)}_{l}}\,,
%     \quad
%     \textnormal{by regularity conditions (covariance bound)}
\\
& \leq &
    \underset{k\,\neq\,l}{\underset{k,l\,\in\,U}{\sum\sum}}\;
    \dfrac{C \cdot n_{\nu}}{N_{\nu}^{2}}
    \cdot
    \left(\,\dfrac{1}{\lambda}\,\dfrac{N_{\nu}}{n_{\nu}}\,\right)^{2}
    \left\vert\,y^{(\nu)}_{k}\,\right\vert
    \cdot
    \left\vert\,y^{(\nu)}_{l}\,\right\vert\,,
\\
&&
    \textnormal{by the Sampling Covariance and Design Weight regularity conditions}
\\
& = &
    \dfrac{C}{\lambda^{2} \cdot n_{\nu}}
    \cdot
    \underset{k\,\neq\,l}{\underset{k,l\,\in\,U}{\sum\sum}}\;
    \left\vert\,y^{(\nu)}_{k}\,\right\vert
    \cdot
    \left\vert\,y^{(\nu)}_{l}\,\right\vert
\;\; \leq \;\;
    \dfrac{C}{\lambda^{2} \cdot n_{\nu}}
    \cdot
    \underset{k,l\,\in\,U}{\sum\sum}\;
    \left\vert\,y^{(\nu)}_{k}\,\right\vert
    \cdot
    \left\vert\,y^{(\nu)}_{l}\,\right\vert
\\
& = &
    \dfrac{C}{\lambda^{2} \cdot n_{\nu}}
    \cdot
    \left(\,
        \underset{k\,\in\,U}{\sum}\,
        \left\vert\,y^{(\nu)}_{k}\,\right\vert
        \,\right)^{2}
\;\; = \;\;
    \dfrac{C \cdot N_{\nu}^{2}}{\lambda^{2} \cdot n_{\nu}}
    \cdot
    \left(\,
        \dfrac{1}{N_{\nu}}
        \cdot
        \underset{k\,\in\,U}{\sum}\,
        \left\vert\,y^{(\nu)}_{k}\,\right\vert
        \,\right)^{2}
\\
& \leq &
    \dfrac{C \cdot N_{\nu}^{2}}{\lambda^{2} \cdot n_{\nu}}
    \cdot
    \left(\,
        \dfrac{1}{N_{\nu}}
        \cdot
        \underset{k\,\in\,U}{\sum}\,
        \left\vert\,y^{(\nu)}_{k}\,\right\vert^{2}
        \,\right),
    \quad
    \textnormal{by Jensen's inequality}
\\
& \leq &
    \dfrac{C \cdot N_{\nu}^{2}}{\lambda^{2} \cdot n_{\nu}}
    \cdot
    M\,,
    \quad
    \textnormal{by the Moment regularity condition},
\end{eqnarray*}
from which it follows that
\begin{equation*}
\left\vert\,\begin{array}{c}
    \textnormal{second summand of}
    \\
    \Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)
    \end{array}\,\right\vert
\;\; = \;\;
    \left\vert\;\;
        \underset{k\,\neq\,l}{\underset{k,l\,\in\,U}{\sum\sum}}
        \left(\,
            \pi^{(\nu)}_{kl}
            \, - \,
            \pi^{(\nu)}_{k}\pi^{(\nu)}_{l}
            \,\right)
        \dfrac{y^{(\nu)}_{k}}{\pi^{(\nu)}_{k}}
        \,
        \dfrac{y^{(\nu)}_{l}}{\pi^{(\nu)}_{l}}
        \;\;\right\vert
\;\; = \;\;
    O\!\left(\,
        \dfrac{N_{\nu}^{2}}{n_{\nu}}
        \,\right)
\end{equation*}
Claim 1 now follows immediately from the preceding two observations.
This completes the proof of Claim 1.

\vskip 0.5cm
\noindent
\textbf{Claim 2:}\quad
$
P\!\left(\;
    \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,-\,Y_{\nu}\,\right\vert
    \,\geq\,
    \delta\cdot\dfrac{N_{\nu}}{\sqrt{n_{\nu}}}
    \;\right)
\;\;\leq\;\;
    \dfrac{
        {\color{white}.}
        \Var\!\left(\,\widehat{Y}_{\nu}^{\textnormal{(HT)}}\,\right)
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\delta^{2} \cdot N_{\nu}^{2}\,\slash\,n_{\nu}}
        }
$,\;\;
for each $\delta > 0$,\, $\nu\in\N$.
\vskip 0.2cm
\noindent
Proof of Claim 2:\;\;
Let \,$\nu\in\N$\, and \,$\delta > 0$\, be fixed. 
Recall Chebyshev's inequality:
For any random variable \,$X$\, with finite variance \,$\sigma_{X}^{2}$\,
(hence, finite expectation $\mu_{X}$ as well), we have:
\begin{equation*}
P\!\left(\,
    \left\vert\,X \overset{{\color{white}.}}{-} \mu_{X}\,\right\vert
    \,\geq\,
    k\cdot\sigma_{X}
    \,\right)
\;\; \leq \;\;
    \dfrac{1}{k^{2}}\,,
\quad
\textnormal{for each \,$k > 0$}
\end{equation*}
Now, for each \,$\nu \in \N$,\, choose
\;$
k_{\nu}
\; := \;
    \dfrac{\delta}{\sqrt{\Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)}}
    \cdot
    \dfrac{N_{\nu}}{\sqrt{n_{\nu}}}
$.\;
It follows that:
\;$
k_{\nu}\cdot\sqrt{\Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)}
\; = \;
    \dfrac{\delta \cdot N_{\nu}}{\sqrt{n_{\nu}}}
$.\;
Hence,
\begin{eqnarray*}
P\!\left(\;
    \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,-\,Y_{\nu}\,\right\vert
    \,\geq\,
    \delta\cdot\dfrac{N_{\nu}}{\sqrt{n_{\nu}}}
    \;\right)
& = &
    P\!\left(\;
        \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,-\,Y_{\nu}\,\right\vert
        \,\geq\,
        k_{\nu}\cdot\sqrt{\Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)}
        \;\right)
\\
& \leq &
    \dfrac{1}{{\color{white}.}k_{\nu}^{2}{\color{white}.}}\,,
    \quad
    \textnormal{by Chebyshev's inequality}
\\
& = &
    \dfrac{
        {\color{white}.}
        n_{\nu} \cdot \Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\delta^{2} \cdot N_{\nu}^{2}}
        }
\;\; = \;\;
    \dfrac{
        {\color{white}.}
        \Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\delta^{2} \cdot N_{\nu}^{2} \,\slash\, n_{\nu}}
        }
\end{eqnarray*}
This completes the proof of Claim 2.

\vskip 0.5cm
\noindent
\textbf{Claim 3:}\quad
There exists \,$K > 0$\, such that
\;$
P\!\left(\;
    \dfrac{
        {\color{white}.}
        \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,-\,Y_{\nu}\,\right\vert
        {\color{white}.}
        }{
        N_{\nu}
        }
    \,<\,
    \dfrac{\delta}{\sqrt{n_{\nu}}}
    \;\right)
\; \geq \;
    1
    \,-\,
    \dfrac{
        {\color{white}.}
        K
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\delta^{2}}
        }
$,\;\;
for each $\delta > 0$,\, $\nu\in\N$.
\vskip 0.2cm
\noindent
Proof of Claim 3:\;\;
This follows from Claim 1 and Claim 2.
Indeed, by Claim 1, there exists \,$K$\, such that
\begin{equation*}
\Var\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,\right)
\;\leq\;
    K\cdot\dfrac{N_{\nu}^{2}}{n_{\nu}}\,,
\quad
\textnormal{for each \,$\nu \in \N$}
\end{equation*}
Hence, by Claim 2, we have, for each \,$\delta > 0$\, and each \,$\nu \in \N$,\,
\begin{equation*}
P\!\left(\;
    \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,-\,Y_{\nu}\,\right\vert
    \,\geq\,
    \delta\cdot\dfrac{N_{\nu}}{\sqrt{n_{\nu}}}
    \;\right)
\;\;\leq\;\;
    \dfrac{
        {\color{white}.}
        \Var\!\left(\,\widehat{Y}_{\nu}^{\textnormal{(HT)}}\,\right)
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\delta^{2} \cdot N_{\nu}^{2}\,\slash\,n_{\nu}}
        }
\;\;\leq\;\;
    \dfrac{
        {\color{white}.}
        K \cdot N_{\nu}^{2} \,\slash\, n_{\nu} 
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\delta^{2} \cdot N_{\nu}^{2}\,\slash\,n_{\nu}}
        }
\;\;\leq\;\;
    \dfrac{
        {\color{white}.}
        K
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\delta^{2}}
        }\,,
\end{equation*}
which is equivalent to:
\begin{equation*}
1
\;-\;
P\!\left(\;
    \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,-\,Y_{\nu}\,\right\vert
    \,<\,
    \delta\cdot\dfrac{N_{\nu}}{\sqrt{n_{\nu}}}
    \;\right)
\;\; \leq \;\;
    \dfrac{
        {\color{white}.}
        K
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\delta^{2}}
        }\,,
\end{equation*}
which is in turn equivalent to:
\begin{equation*}
P\!\left(\;
    \dfrac{
        {\color{white}.}
        \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,-\,Y_{\nu}\,\right\vert
        {\color{white}.}
        }{
        N_{\nu}
        }
    \,<\,
    \dfrac{\delta}{\sqrt{n_{\nu}}}
    \;\right)
\;\; \geq \;\;
    1
    \;-\;
    \dfrac{
        {\color{white}.}
        K
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\delta^{2}}
        }
\end{equation*}
This completes the proof of Claim 3.

\vskip 0.5cm
\noindent
The Proposition now follows readily from Claim 3.
Let \,$\varepsilon > 0$\, be given.
Choose \,$\delta(\varepsilon) > 0$\, such that \,$\dfrac{K}{\delta(\varepsilon)^{2}} \,<\, \varepsilon$.\,
By Claim 3, we have:
\begin{equation*}
P\!\left(\;
    \dfrac{
        {\color{white}.}
        \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,-\,Y_{\nu}\,\right\vert
        {\color{white}.}
        }{
        N_{\nu}
        }
    \,<\,
    \dfrac{\delta(\varepsilon)}{\sqrt{n_{\nu}}}
    \;\right)
% \;\; = \;\;
% P\!\left(\;
%     \left\vert\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}\,-\,Y_{\nu}\,\right\vert
%     \,<\,
%     \delta(\varepsilon)\cdot\dfrac{N_{\nu}}{\sqrt{n_{\nu}}}
%     \;\right)
\;\; \geq \;\;
    1
    \,-\,
    \dfrac{
        {\color{white}.}
        K
        {\color{white}.}
        }{
        \overset{{\color{white}.}}{\delta(\varepsilon)^{2}}
        }
\;\; > \;\;
    1 \,-\, \varepsilon\,,
\end{equation*}
as required.
This completes the proof of the Proposition.
\qed

\subsection{Some asymptotics of the ratio estimator}

          %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

\vskip 0.5cm
\begin{theorem}
\mbox{}
\vskip 0.05cm
\noindent
\textbf{Setting}
\vskip 0.1cm
\noindent
Suppose:
\begin{itemize}
\item
    $\left\{\;\overset{{\color{white}.}}{U_{\nu}}\;\right\}_{\nu=1}^{\infty}$\,
    is a sequence of (nonempty) finite populations whose sequence
    $\left\{\,N_{\nu} \overset{{\color{white}-}}{:=} \vert\,U_{\nu}\,\vert\,\right\}_{\nu=1}^{\infty}$\,
    of population sizes tends to infinity as $\nu \rightarrow \infty$, i.e.,
    \,$\underset{\nu\rightarrow\infty}{\lim}\,N_{\nu} \,=\, \infty$.\,
\item
    For each \,$\nu \in \N$,\,
    \,$p_{\nu} : \mathcal{P}(U_{\nu}) \longrightarrow [\,0,1)$\,
    is a {\color{red}fixed-sample-size} sampling design on \,$U_{\nu}$\,
    such that
    \begin{equation*}
    \pi^{(\nu)}_{i}
    \;\; := \;\;
        \underset{s\,\ni\,i}{\sum}\;\,
        p_{\nu}\!\left(\,s\,\right)
    \;\; > \;\;
        0\,,
    \quad
    \textnormal{for each \,$i \in U_{\nu}$,\, and for each \,$\nu\in\N$}
    \end{equation*}
\item
    For each \,$\nu \in \N$,\,
    \,$x^{(\nu)} : U_{\nu} \longrightarrow \Re$\,
    and
    \,$y^{(\nu)} : U_{\nu} \longrightarrow \Re$\,
    are two $\Re$-valued population characteristic defined on \,$U_{\nu}$.\,
\end{itemize}
\vskip 0.2cm
\noindent
\textbf{Notations}
\begin{itemize}
\item
    For each \,$\nu\in\N$,\,
    \,$n_{\nu}$\, is the fixed sample size of the sampling design \,$p_{\nu}$, i.e.,\,
    \begin{equation*}
    n_{\nu}
    \;\; := \;\;
        \underset{k\,\in\,U_{\nu}}{\sum}\,I^{(\nu)}_{i}(s)\,,
    \quad
    \textnormal{for each sample \,$s \subset U_{\nu}$\, with \,$p_{\nu}(s) > 0$}
    \end{equation*}
    Note that left-hand side \,$n_{\nu}$\, is indeed well-defined
    since the right-hand side is independent of the sample \,$s$ (as long as $p_{\nu}(s) > 0$),\,
    by the fixed-sample-size assumption on the sampling design \,$p_{\nu}$.
\item
    For each \,$\nu\in\N$,\,
    \,$X_{\nu}$\, and \,$Y_{\nu}$\,
    are respectively the population totals of
    \,$x^{(\nu)}$\, and \,$y^{(\nu)}$,\, i.e.,
    \begin{equation*}
    X_{\nu}
    \;\, := \,
        \underset{i \in U_{\nu}}{\sum}\,x^{(\nu)}_{i}\,,
    \quad\quad
    \textnormal{and}
    \quad\quad
    Y_{\nu}
    \;\, := \,
        \underset{i \in U_{\nu}}{\sum}\,y^{(\nu)}_{i}
    \end{equation*}
    Furthermore, for each \,$\nu \in \N$,\, we define:
    \begin{equation*}
    R_{\nu}
    \;\; := \;\;
        \dfrac{
            {\color{white}.}
            Y_{\nu}
            {\color{white}.}
            }{
            X_{\nu}
            }
    \end{equation*}
\item
    For each \,$\nu\in\N$,\,
    \,$\widehat{X}^{\textnormal{(HT)}}_{\nu}$\,
    and
    \,$\widehat{Y}^{\textnormal{(HT)}}_{\nu}$\,
    are respectively the Horvitz-Thompson estimators for the population totals
    \,$X_{\nu}$\, and \,$Y_{\nu}$,\, i.e.,
    \begin{equation*}
    \widehat{X}^{\textnormal{(HT)}}_{\nu}(\,s\,)
    \;\; := \;\;
        \underset{i \in s}{\sum}\;
        \dfrac{ x^{(\nu)}_{i} }{ \pi^{(\nu)}_{i} }\,,
    \quad\quad
    \textnormal{and}
    \quad\quad
    \widehat{Y}^{\textnormal{(HT)}}_{\nu}(\,s\,)
    \;\; := \;\;
        \underset{i \in s}{\sum}\;
        \dfrac{ y^{(\nu)}_{i} }{ \pi^{(\nu)}_{i} }\,,
    \quad\quad
    \textnormal{for each \,$s \subset U_{\nu}$\, with \,$p_{\nu}(s) > 0$}
    \end{equation*}
    Furthermore, for each \,$\nu \in \N$,\, we define:
    \begin{equation*}
    \widehat{R}_{\nu}(\,s\,)
    \;\; := \;\;
        \dfrac{
            {\color{white}.}
            \widehat{Y}^{\textnormal{(HT)}}_{\nu}(\,s\,)
            {\color{white}.}
            }{
            \widehat{X}^{\textnormal{(HT)}}_{\nu}(\,s\,)
            }\,,
    \quad\quad
    \textnormal{for each \,$s \subset U_{\nu}$\, with \,$p_{\nu}(s) > 0$}
    \end{equation*}
\end{itemize}
\vskip 0.2cm
\noindent
\textbf{Hypotheses (``regularity conditions'')}
\vskip 0.1cm
\noindent
Suppose the entities above satisfy the following hypotheses (``regularity conditions''):
\begin{itemize}
\item
    Design weight condition:
    There exists \,$\lambda > 0$\,
    such that
    \begin{equation*}
    \dfrac{1}{\pi^{(\nu)}_{i}}
    \;\leq\;
        \dfrac{1}{\lambda}
        \cdot
        \dfrac{N_{\nu}}{n_{\nu}}\,,
    \quad
    \textnormal{for each \,$i \in U_{\nu}$,\; and for each \,$\nu \in \N$}
    \end{equation*}
\item
    Sampling covariance condition:
    There exists a constant \,$C > 0$\, such that,
    for all \,$i \neq j$,\,
    \begin{equation*}
    \left\vert\; \pi^{(\nu)}_{ij} - \pi^{(\nu)}_{i}\,\pi^{(\nu)}_{j} \;\right\vert
    \;\; \leq \;\;
        C\cdot\dfrac{n_{\nu}}{N_{\nu}^{2}}\,,
    \quad
    \textnormal{for each \,$i,j \in U_{\nu}$ with $i \neq j$,\, for each \,$\nu\in\N$}
    \end{equation*}
\item
    Moment condition:\;
    There exists \,$M > 0$\, such that
    \begin{equation*}
    \max\left\{\;
        \dfrac{1}{N_{\nu}}\cdot\underset{i\,\in\,U_{\nu}}{\sum}\!\left(\,x^{(\nu)}_{i}\,\right)^{2}
        \,,\;
        \dfrac{1}{N_{\nu}}\cdot\underset{i\,\in\,U_{\nu}}{\sum}\!\left(\,y^{(\nu)}_{i}\,\right)^{2}
        \;\right\}
    \;\; \leq \;\;M \;\; < \;\; \infty\,,
    \quad
    \textnormal{for each \,$\nu\in\N$}
    \end{equation*}
\item
    Denominator positivity condition:\;
    There exists \,$c > 0$\, such that
    \begin{equation*}
    \dfrac{
        {\color{white}.}
        \left\vert\,X_{\nu}\,\right\vert
        {\color{white}.}
        }{
        N_{\nu}
        }
    \;\; \geq \;\; c \;\; > \;\; 0\,,
    \quad
    \textnormal{for each \,$\nu\in\N$}
    \end{equation*}
\end{itemize}
\vskip 0.2cm
\noindent
\textbf{Conclusions}
\vskip 0.1cm
\noindent
Then, the following statements are true:
\begin{enumerate}
\item
    The first-order Taylor approximation at
    \,$(X_{\nu},Y_{\nu})$\,
    of
    \;$
    \widehat{R}_{\nu}
    \, := \,
        \dfrac{
            \widehat{Y}^{\textnormal{(HT)}}_{\nu}
            }{
            \widehat{X}^{\textnormal{(HT)}}_{\nu}
            }
    $\;
    is given by:
    \begin{eqnarray*}
    L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
    & = &
        R_{\nu}
        {\color{white}.}
        \; - \;
        \dfrac{Y_{\nu}}{X_{\nu}^{2}}\!\left(\,\widehat{X}^{\textnormal{(HT)}}_{\nu} - X_{\nu} \,\right)
        \; + \;
        \dfrac{1}{X_{\nu}}\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu} - Y_{\nu} \,\right)
    \\
    & = &
        \dfrac{ Y_{\nu} }{ X_{\nu} }
        \; + \;
        \dfrac{1}{X_{\nu}}
        \cdot
        \left(\;
            \widehat{Y}^{\textnormal{(HT)}}_{\nu}
            \,-\,
            \dfrac{ Y_{\nu} }{ X_{\nu} }
            \,
            \widehat{X}^{\textnormal{(HT)}}_{\nu}
            \,\right)
    \end{eqnarray*}
\item
    The residual of the first-order Taylor approximation at
    \,$(X_{\nu},Y_{\nu})$\,
    of
    \;$
    \widehat{R}_{\nu}
    \, := \,
        \dfrac{
            \widehat{Y}^{\textnormal{(HT)}}_{\nu}
            }{
            \widehat{X}^{\textnormal{(HT)}}_{\nu}
            }
    $\;
    is of stochastic order
    \,$O_{p}\!\left(\,\dfrac{1}{n_{\nu}}\,\right)$;\,
    more precisely,
    \begin{eqnarray*}
    \widehat{R}_{\nu}
    & = &
        L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
        \; + \;
        O_{p}\!\left(\,\dfrac{1}{n_{\nu}}\,\right)
    \\
    & = &
        \dfrac{ Y_{\nu} }{ X_{\nu} }
        \; + \;
        \dfrac{1}{X_{\nu}}
        \cdot
        \left(\;
            \widehat{Y}^{\textnormal{(HT)}}_{\nu}
            \,-\,
            \dfrac{ Y_{\nu} }{ X_{\nu} }
            \,
            \widehat{X}^{\textnormal{(HT)}}_{\nu}
            \,\right)
        % \; - \;
        % \dfrac{Y_{\nu}}{X_{\nu}^{2}}\!\left(\,\widehat{X}^{\textnormal{(HT)}}_{\nu} - X_{\nu} \,\right)
        % \; + \;
        % \dfrac{1}{X_{\nu}}\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu} - Y_{\nu} \,\right)
        \; + \;
        O_{p}\!\left(\,\dfrac{1}{n_{\nu}}\,\right)
    \end{eqnarray*}
\item
    The sequence of differences
    \;$
    \MSE\!\left(\,\widehat{R}_{\nu}\,\right)
    \,-\,
    \Var\!\left(\,L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)\,\right)
    $\;
    is of stochastic order
    \,$O_{p}\!\left(\,\dfrac{1}{n_{\nu}}\,\right)$;\,
    more precisely,
    \begin{equation*}
    \MSE\!\left(\,\widehat{R}_{\nu}\,\right)
    \;\; = \;\;
        \Var\!\left(\,L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)\,\right)
        \; + \;
        O\!\left(\,\dfrac{1}{n_{\nu}^{2}}\,\right)
    \end{equation*}
\end{enumerate}
\end{theorem}

          %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

\vskip 0.5cm
\noindent
\proof
\begin{enumerate}
\item
    Let \,$f(x,y) \,:=\, \dfrac{y}{x}$.\,
    Then,
    \;\;$
    \dfrac{\partial\,f}{\partial x}(x,y)
    \, = \,
        -\,\dfrac{y}{x^{2}}\,,
    \;\,
    \textnormal{and}
    \;\;
    \dfrac{\partial\,f}{\partial y}(x,y)
    \, = \,
        +\,\dfrac{1}{x}
    $.\;\;
    Hence, the first-order Taylor approximation at \,$(x_{0},y_{0})$\, of \,$f(x,y)$\, is given by:
    \begin{eqnarray*}
    f(x,y)
    & \approx &
        f(x_{0},y_{0})
        \; + \;
        \dfrac{\partial\,f}{\partial x}(x_{0},y_{0})\cdot\left(\,x - x_{0}\,\right)
        \; + \;
        \dfrac{\partial\,f}{\partial y}(x_{0},y_{0})\cdot\left(\,y - y_{0}\,\right)
    \\
    & = &
        \dfrac{ y_{0} }{ x_{0} }
        \; - \;
        \dfrac{y_{0}}{x_{0}^{2}}\cdot\left(\,x - x_{0}\,\right)
        \; + \;
        \dfrac{1}{x_{0}}\cdot\left(\,y - y_{0}\,\right)
    \end{eqnarray*}
    The observation immediately implies that the first-order Taylor approximation at
    \,$(X_{\nu},Y_{\nu})$\,
    of
    \;$
    \widehat{R}_{\nu}
    \, := \,
        \dfrac{
            \widehat{Y}^{\textnormal{(HT)}}_{\nu}
            }{
            \widehat{X}^{\textnormal{(HT)}}_{\nu}
            }
    $\;
    is given by:
    \begin{eqnarray*}
    L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
    & = &
        R_{\nu}
        {\color{white}.}
        \; - \;
        \dfrac{Y_{\nu}}{X_{\nu}^{2}}\!\left(\,\widehat{X}^{\textnormal{(HT)}}_{\nu} - X_{\nu} \,\right)
        \; + \;
        \dfrac{1}{X_{\nu}}\!\left(\,\widehat{Y}^{\textnormal{(HT)}}_{\nu} - Y_{\nu} \,\right)
    \\
    & = &
        \dfrac{ Y_{\nu} }{ X_{\nu} }
        \; + \;
        \dfrac{1}{X_{\nu}}
        \cdot
        \left(\;
            \widehat{Y}^{\textnormal{(HT)}}_{\nu}
            \,-\,
            \dfrac{ Y_{\nu} }{ X_{\nu} }
            \,
            \widehat{X}^{\textnormal{(HT)}}_{\nu}
            \,\right)
    \end{eqnarray*}
\item
    First, we state and prove two claims.

    \vskip 0.3cm
    \noindent
    \textbf{Claim 1:}\quad
    $\dfrac{X_{\nu}}{N_{\nu}} \,=\, O(1)$\;\,
    and
    \;\,$\dfrac{Y_{\nu}}{N_{\nu}} \,=\, O(1)$\,
    \vskip 0.1cm
    \noindent
    Proof of Claim 1:\;\;
    By the Cauchy-Schwarz inequality, we have:
    \begin{eqnarray*}
    \left\vert\;
        \underset{k\,\in\,U_{\nu}}{\sum}\,
        y^{(\nu)}_{k} \cdot 1
        \;\right\vert^{2}
    \;\; \leq \;\;
        \left(\;
            \underset{k\,\in\,U_{\nu}}{\sum}
            (y^{(\nu)}_{k})^{2}
            \;\right)
        \cdot
        \left(\;
            \underset{k\,\in\,U_{\nu}}{\sum}
            (\,1\,)^{2}
            \;\right)
    \;\; = \;\;
        N_{\nu}
        \cdot
        \left(\;
            \underset{k\,\in\,U_{\nu}}{\sum}
            (y^{(\nu)}_{k})^{2}
            \;\right),
    \end{eqnarray*}
    which in turn implies
    \begin{eqnarray*}
    \left\vert\;\,
        \dfrac{Y_{\nu}}{N_{\nu}}
        \;\right\vert^{2}
    \;\; = \;\;
        \left\vert\;\,
            \dfrac{1}{N_{\nu}}
            \cdot
            \underset{k\,\in\,U_{\nu}}{\sum}\,
            y^{(\nu)}_{k}
            \;\right\vert^{2}
    \;\; \leq \;\;
        \dfrac{1}{N_{\nu}}
        \cdot
        \underset{k\,\in\,U_{\nu}}{\sum}
        (y^{(\nu)}_{k})^{2}
    \;\; \leq \;\;
        M\,,
    \end{eqnarray*}
    where the last inequality follows by the Moment regularity condition.
    We therefore have:
    \begin{eqnarray*}
    \left\vert\;\,
        \dfrac{Y_{\nu}}{N_{\nu}}
        \;\right\vert
    \;\; \leq \;\;
        \sqrt{{\color{white}.}M{\color{white}.}}\,;
    \end{eqnarray*}
    equivalently,
    \,$
    \dfrac{Y_{\nu}}{N_{\nu}} \,=\, O(1)
    $.\,
    Similarly for \,$X_{\nu} \slash N_{\nu}$.\,
    This completes the proof of Claim 1.

    \vskip 0.3cm
    \noindent
    \textbf{Claim 2:}\quad
    $
    \dfrac{
        1
        }{
        (\left. X_{\nu}\right\slash N_{\nu}\,)^{2}
        \cdot
        \overset{{\color{white}.}}{
            \left(\left.\widehat{X}^{\textnormal{(HT)}}_{\nu}\right\slash N_{\nu}\right)
            }            
        }
    \; = \;
        O_{p}(1)
    $
    \vskip 0.1cm
    \noindent
    Proof of Claim 2:\;\;
    Recall that, by Theorem \ref{AsymptoticsHTForDomainTotal}, we have
    \;$
    \dfrac{
        {\color{white}.}
        \widehat{X}^{\textnormal{(HT)}}_{\nu}
        \, - \,
        X_{\nu}
        {\color{white}.}
        }{
        N_{\nu}
        }
    \, = \,
    O_{p}\!\left(\,\dfrac{1}{\sqrt{n_{\nu}}}\,\right)
    $;\;
    hence,
    \begin{equation*}
    \dfrac{
        {\color{white}.}
        \widehat{X}^{\textnormal{(HT)}}_{\nu}
        {\color{white}.}
        }{
        N_{\nu}
        }
    \;\; = \;\;
    \dfrac{
        {\color{white}.}
        X_{\nu}
        {\color{white}.}
        }{
        N_{\nu}
        }
    \; + \;
    O_{p}\!\left(\,\dfrac{1}{\sqrt{n_{\nu}}}\,\right)
    \end{equation*}
    which in turn implies
    \begin{eqnarray*}
    \left(\,
        \dfrac{X_{\nu}}{N_{\nu}}
        \,\right)^{\!2}
    \cdot
    \left(\,\dfrac{
        {\color{white}.}
        \widehat{X}^{\textnormal{(HT)}}_{\nu}
        {\color{white}.}
        }{
        N_{\nu}
        }\,\right)
    & = &
        \left(\,
            \dfrac{X_{\nu}}{N_{\nu}}
            \,\right)^{\!2}
        \cdot
        \dfrac{
            {\color{white}.}
            X_{\nu}
            {\color{white}.}
            }{
            N_{\nu}
            }
        \; + \;
        \left(\,
            \dfrac{X_{\nu}}{N_{\nu}}
            \,\right)^{\!2}
        \cdot
        O_{p}\!\left(\,\dfrac{1}{\sqrt{n_{\nu}}}\,\right)
    \;\; = \;\;
        \left(\,
            \dfrac{X_{\nu}}{N_{\nu}}
            \,\right)^{\!3}
        \; + \;
        O\!\left(\,1\,\right)
        \cdot
        O_{p}\!\left(\,\dfrac{1}{\sqrt{n_{\nu}}}\,\right)
    \\
    & = &
        \left(\,
            \dfrac{X_{\nu}}{N_{\nu}}
            \,\right)^{\!3}
        \; + \;
        O_{p}\!\left(\,\dfrac{1}{\sqrt{n_{\nu}}}\,\right)
    \end{eqnarray*}
    Now, recall that by the Denominator Positivity regularity condition,
    \,$\dfrac{X_{\nu}}{N_{\nu}}$\,
    is bounded from below away from zero.
    Hence,
    \,$\left(\,\dfrac{1}{X_{\nu} \slash N_{\nu}}\,\right)^{3}$\,
    is bounded from above, from which Claim 2 follows immediately.
    This completes the proof of Claim 2.

    \vskip 0.5cm
    Next, we introduce the following notation for the residual:
    \begin{equation*}
    \Delta_{\nu}
    \;\; := \;\;
        \widehat{R}_{\nu}
        \; - \;
        L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
    \;\; = \;\;
        \widehat{R}_{\nu}
        \; - \;
        R_{\nu}
        \; - \;
        \dfrac{1}{X_{\nu}}
        \cdot
        \left(\;
            \widehat{Y}^{\textnormal{(HT)}}_{\nu}
            \,-\,
            \dfrac{ Y_{\nu} }{ X_{\nu} }
            \,
            \widehat{X}^{\textnormal{(HT)}}_{\nu}
            \,\right)
    \end{equation*}
    We need to establish:
    \,$\Delta_{\nu} \, = \, O_{p}\!\left(\,\dfrac{1}{n_{\nu}}\,\right)$.\,
    To this end, note:
    \begin{eqnarray*}
    \Delta_{\nu}
    & := &
        \dfrac{
            {\color{white}.}
            \widehat{Y}^{\textnormal{(HT)}}_{\nu}
            {\color{white}.}
            }{
            \widehat{X}^{\textnormal{(HT)}}_{\nu}
            }
        \; - \;
        \dfrac{ Y_{\nu} }{ X_{\nu} }
        \; - \;
        \dfrac{1}{X_{\nu}}
        \cdot
        \left(\;
            \widehat{Y}^{\textnormal{(HT)}}_{\nu}
            \,-\,
            \dfrac{ Y_{\nu} }{ X_{\nu} }
            \,
            \widehat{X}^{\textnormal{(HT)}}_{\nu}
            \,\right)
    \\
    & = &
        \dfrac{
            {\color{white}.}
            X_{\nu}\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}
            \, - \,
            Y_{\nu}\,\widehat{X}^{\textnormal{(HT)}}_{\nu}
            {\color{white}.}
            }{
            \overset{{\color{white}.}}{X_{\nu}\,\widehat{X}^{\textnormal{(HT)}}_{\nu}}            
            }
        \; - \;
        \dfrac{
            {\color{white}.}
            X_{\nu}\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}
            \, - \,
            Y_{\nu}\,\widehat{X}^{\textnormal{(HT)}}_{\nu}
            {\color{white}.}
            }{
            \overset{{\color{white}.}}{X_{\nu}^{2}}
            }
    \\
    & = &
        \left(\,
            \dfrac{
                {\color{white}.}
                X_{\nu}\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}
                \, - \,
                Y_{\nu}\,\widehat{X}^{\textnormal{(HT)}}_{\nu}
                {\color{white}.}
                }{
                \overset{{\color{white}.}}{X_{\nu}}
                }
            \,\right)
        \cdot
        \left(\,
            \dfrac{
                1
                }{
                \overset{{\color{white}.}}{\widehat{X}^{\textnormal{(HT)}}_{\nu}}
                }
            \; - \;
            \dfrac{
                1
                }{
                \overset{{\color{white}.}}{X_{\nu}}
                }
            \,\right)
    \;\; = \;\;
        \left(\,
            \dfrac{
                {\color{white}.}
                X_{\nu}\,\widehat{Y}^{\textnormal{(HT)}}_{\nu}
                \, - \,
                Y_{\nu}\,\widehat{X}^{\textnormal{(HT)}}_{\nu}
                {\color{white}.}
                }{
                \overset{{\color{white}.}}{X_{\nu}}
                }
            \,\right)
        \cdot
        \left(\,
            \dfrac{
                X_{\nu} \,-\, \widehat{X}^{\textnormal{(HT)}}_{\nu}
                }{
                X_{\nu}\,\widehat{X}^{\textnormal{(HT)}}_{\nu}
                }
            \,\right)
    \\
    & = &
        \left(\;
            X_{\nu}
            \cdot
            \left(\,
                \widehat{Y}^{\textnormal{(HT)}}_{\nu}
                \, - \,
                Y_{\nu}
                \,\right)
            \; \overset{{\color{white}1}}{+} \;
            Y_{\nu}
            \cdot
            \left(\,
                X_{\nu}
                \, - \,
                \widehat{X}^{\textnormal{(HT)}}_{\nu}
                \,\right)
            \;\right)
        \cdot
        \left(\,
            \dfrac{
                X_{\nu} \,-\, \widehat{X}^{\textnormal{(HT)}}_{\nu}
                }{
                X_{\nu}^{2}\,\widehat{X}^{\textnormal{(HT)}}_{\nu}
                }
            \,\right)
    \\
    & = &
        \left(\;
            \dfrac{X_{\nu}}{N_{\nu}}
            \cdot
            \left(\,
                \dfrac{
                    \widehat{Y}^{\textnormal{(HT)}}_{\nu}
                    \, - \,
                    Y_{\nu}
                    }{
                    N_{\nu}
                    }
                \,\right)
            \; \overset{{\color{white}1}}{+} \;
            \dfrac{Y_{\nu}}{N_{\nu}}
            \cdot
            \left(\,
                \dfrac{
                    X_{\nu}
                    \, - \,
                    \widehat{X}^{\textnormal{(HT)}}_{\nu}
                    }{
                    N_{\nu}
                    }
                \,\right)
            \;\right)
        \cdot
        \left(\,
            \dfrac{
                \left.\left(\,X_{\nu} \,-\, \widehat{X}^{\textnormal{(HT)}}_{\nu}\,\right)\,\right\slash N_{\nu}
                }{
                (X_{\nu}\slash N_{\nu})^{2}\,
                \left(\left.\widehat{X}^{\textnormal{(HT)}}_{\nu}\right\slash N_{\nu}\right)
                }
            \,\right)
    \\
    & = &
        \left(\;
            O(1) \cdot O_{p}\!\left(\dfrac{1}{\sqrt{n_{\nu}}}\right)
            \; + \;
            O(1) \cdot O_{p}\!\left(\dfrac{1}{\sqrt{n_{\nu}}}\right)
            \;\right)
        \cdot
        O_{p}\!\left(\dfrac{1}{\sqrt{n_{\nu}}}\right)
        \cdot
        O_{p}(1)\,,
        \;\;
        \textnormal{by Claim 1, Claim 2, and Theorem \ref{AsymptoticsHTForDomainTotal}}
    \\
    & = &
        O_{p}\!\left(\dfrac{1}{\sqrt{n_{\nu}}}\right)
        \cdot
        O_{p}\!\left(\dfrac{1}{\sqrt{n_{\nu}}}\right)
    \;\; = \;\;
        O_{p}\!\left(\dfrac{1}{n_{\nu}}\right),
    \end{eqnarray*}
    as required.
\item
    \textbf{Claim 1:}\quad
    $\MSE\!\left(\,\widehat{R}_{\nu}\,\right)$\,
    can be decomposed as follows:
    \begin{eqnarray*}
    \MSE\!\left(\,\widehat{R}_{\nu}\,\right)
    & = &
        \Var\!\left[\,
            L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
            \,\right]
        \; + \;
        E\!\left[\;\Delta_{\nu}^{\!2}\;\right]
        \; + \;
        2 \cdot
        E\!\left[\;
            \Delta_{\nu}
            \cdot
            \left(\,
                L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
                \,-\,
                R_{\nu}
                \,\right)
            \;\right]
    \end{eqnarray*}
    \vskip 0.1cm
    \noindent
    Proof of Claim 1:
    \begin{eqnarray*}
    \MSE\!\left(\,\widehat{R}_{\nu}\,\right)
    & := &
        E\!\left[\,\left(\,\widehat{R}_{\nu}\,-\,R_{\nu}\,\right)^{\!2}\,\right]
    \;\; = \;\;
        E\!\left[\,\left(\,
            \widehat{R}_{\nu}
            {\color{blue}
            \;-\;
            L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
            \,+\,
            L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
            }
            \,-\,
            R_{\nu}
            \,\right)^{\!2}\,\right]
    \\
    & = &
        E\!\left[\,\left(\,
            L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)

            R_{\nu}
            \,\right)^{\!2}\,\right]
        \; + \;

            \widehat{R}_{\nu}
            \;-\;
            L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)

    \\
    &&
        + \;
        2 \cdot
        E\!\left[\,
            \left(\,
                \widehat{R}_{\nu}
                \;-\;
                L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
                \,\right)
            \cdot
            \left(\,
                L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
                \,-\,
                R_{\nu}
                \,\right)
            \,\right]
    % \\
    % & = &

    %         L_{(X_{\nu},Y_{\nu})}\!\left(\,\widehat{R}_{\nu}\,\right)
    %         \,\right]
    %     \; + \;
    %     E\!\left[\,\left(\,
    %         \widehat{R}_{\nu}

    & = &

    as required.
    This completes the proof of Claim 1.
